001package org.clafer.choco.constraint.propagator;
002
003import gnu.trove.iterator.TIntIterator;
004import java.util.Arrays;
005import org.clafer.collection.FixedCapacityIntSet;
006import solver.constraints.propagators.Propagator;
007import solver.constraints.propagators.PropagatorPriority;
008import solver.exception.ContradictionException;
009import solver.variables.EventType;
010import solver.variables.IntVar;
011import solver.variables.SetVar;
012import solver.variables.Variable;
013import util.ESat;
014
015/**
016 * <p>
017 * Join a unary set relation with a binary function. This propagator is a
018 * specialized version of {@link PropJoinRelation}. The {@code take} variable is
019 * the unary relation and the {@code ref} variables are the function. The
020 * {@code to} variable is the result of the join.
021 * </p>
022 * <p>
023 * Here is how the binary function is encoded. Consider the function:
024 * {@code (0, 1), (1, 3), (2, 1)}. This is encoded as 3 different {@code ref}
025 * variables: {@code ref0={1}, ref1={3}, ref2={1}}.
026 * </p>
027 *
028 * @author jimmy
029 */
030public class PropJoinFunction extends Propagator<Variable> {
031
032    private final SetVar take;
033    private final IntVar[] refs;
034    private final SetVar to;
035
036    public PropJoinFunction(SetVar take, IntVar[] refs, SetVar to) {
037        super(buildArray(take, to, refs), PropagatorPriority.LINEAR, true);
038        this.take = take;
039        this.refs = refs;
040        this.to = to;
041    }
042
043    private static Variable[] buildArray(SetVar take, SetVar to, IntVar[] refs) {
044        Variable[] array = new Variable[refs.length + 2];
045        array[0] = take;
046        array[1] = to;
047        System.arraycopy(refs, 0, array, 2, refs.length);
048        return array;
049    }
050
051    private boolean isTakeVar(int idx) {
052        return idx == 0;
053    }
054
055    private boolean isToVar(int idx) {
056        return idx == 1;
057    }
058
059    private boolean isRefVar(int idx) {
060        return idx >= 2;
061    }
062
063    private int getRefVarIndex(int idx) {
064        assert isRefVar(idx);
065        return idx - 2;
066    }
067
068    @Override
069    public int getPropagationConditions(int vIdx) {
070        if (isTakeVar(vIdx)) {
071            return EventType.ADD_TO_KER.mask + EventType.REMOVE_FROM_ENVELOPE.mask;
072        }
073        if (isToVar(vIdx)) {
074            return EventType.ADD_TO_KER.mask + EventType.REMOVE_FROM_ENVELOPE.mask;
075        }
076        assert isRefVar(vIdx);
077        return EventType.INT_ALL_MASK();
078    }
079
080    private boolean findMate(int toEnv) throws ContradictionException {
081        boolean inKer = to.kernelContains(toEnv);
082        int mate = -1;
083        for (int j = take.getEnvelopeFirst(); j != SetVar.END; j = take.getEnvelopeNext()) {
084            if (refs[j].contains(toEnv)) {
085                // Found a second mate.
086                if (mate != -1 || !inKer) {
087                    mate = -2;
088                    break;
089                }
090                mate = j;
091            }
092        }
093        if (mate == -1) {
094            // No mates.
095            to.removeFromEnvelope(toEnv, aCause);
096        } else if (mate != -2 && inKer) {
097            // One mate.
098            take.addToKernel(mate, aCause);
099            return refs[mate].instantiateTo(toEnv, aCause);
100        }
101        return false;
102    }
103
104    private void findMates() throws ContradictionException {
105        for (int i = to.getEnvelopeFirst(); i != SetVar.END; i = to.getEnvelopeNext()) {
106            if (findMate(i)) {
107                findMates();
108                return;
109            }
110        }
111    }
112
113    private int pruneRefsPickTo(boolean forceNoMoreDuplicates) throws ContradictionException {
114        int sameRefs = 0;
115        FixedCapacityIntSet set = new FixedCapacityIntSet(take.getKernelSize());
116        // Prune refs, Pick to
117        for (int i = take.getKernelFirst(); i != SetVar.END; i = take.getKernelNext()) {
118            PropUtil.intSubsetEnv(refs[i], to, aCause);
119            if (refs[i].instantiated()) {
120                int value = refs[i].getValue();
121                sameRefs += set.add(value) ? 0 : 1;
122                to.addToKernel(value, aCause);
123            }
124        }
125        for (int i = take.getKernelFirst(); i != SetVar.END; i = take.getKernelNext()) {
126            if (!refs[i].instantiated()) {
127                if (PropUtil.isDomainSubsetOf(refs[i], set)) {
128                    sameRefs++;
129                } else if (forceNoMoreDuplicates) {
130                    TIntIterator iter = set.iterator();
131                    while (iter.hasNext()) {
132                        refs[i].removeValue(iter.next(), aCause);
133                    }
134                    if (refs[i].instantiated()) {
135                        contradiction(refs[i], "Take too small");
136                    }
137                }
138            }
139        }
140        return sameRefs;
141    }
142
143    @Override
144    public void propagate(int evtmask) throws ContradictionException {
145        // Prune to
146        findMates();
147
148        // Prune refs, Pick to
149        int sameRefs = pruneRefsPickTo(false);
150
151        // Prune take
152        for (int i = take.getEnvelopeFirst(); i != SetVar.END; i = take.getEnvelopeNext()) {
153            if (!PropUtil.domainIntersectEnv(refs[i], to)) {
154                take.removeFromEnvelope(i, aCause);
155            }
156        }
157
158        int minTakeSize = sameRefs + to.getKernelSize();
159        if (minTakeSize == take.getEnvelopeSize()) {
160            take.instantiateTo(PropUtil.iterateEnv(take), aCause);
161            to.instantiateTo(PropUtil.iterateKer(to), aCause);
162            sameRefs = pruneRefsPickTo(true);
163        }
164        minTakeSize = sameRefs + to.getKernelSize();
165        if (minTakeSize > take.getEnvelopeSize()) {
166            contradiction(take, "Take too large");
167        }
168    }
169
170    @Override
171    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
172        forcePropagate(EventType.FULL_PROPAGATION);
173    }
174
175    @Override
176    public ESat isEntailed() {
177        for (int i = take.getKernelFirst(); i != SetVar.END; i = take.getKernelNext()) {
178            if (!PropUtil.domainIntersectEnv(refs[i], to)) {
179                return ESat.FALSE;
180            }
181        }
182        if (!take.instantiated() || !to.instantiated()) {
183            return ESat.UNDEFINED;
184        }
185        FixedCapacityIntSet values = new FixedCapacityIntSet(refs.length);
186        for (IntVar ref : refs) {
187            if (!ref.instantiated()) {
188                return ESat.UNDEFINED;
189            }
190            values.add(ref.getValue());
191        }
192        return values.containsAll(to.getValue()) ? ESat.TRUE : ESat.UNDEFINED;
193    }
194
195    @Override
196    public String toString() {
197        return "joinFunction(" + take + ", " + Arrays.toString(refs) + ", " + to + ")";
198    }
199}
