001package org.clafer.choco.constraint.propagator;
002
003import java.util.Arrays;
004import org.clafer.collection.FixedCapacityIntSet;
005import solver.constraints.Propagator;
006import solver.constraints.PropagatorPriority;
007import solver.exception.ContradictionException;
008import solver.variables.EventType;
009import solver.variables.IntVar;
010import solver.variables.SetVar;
011import solver.variables.Variable;
012import solver.variables.delta.monitor.SetDeltaMonitor;
013import util.ESat;
014import util.procedure.IntProcedure;
015
016/**
017 * <p>
018 * Join a unary set relation with a binary function. This propagator is a
019 * specialized version of {@link PropJoinRelation}. The {@code take} variable is
020 * the unary relation and the {@code ref} variables are the function. The
021 * {@code to} variable is the result of the join.
022 * </p>
023 * <p>
024 * Here is how the binary function is encoded. Consider the function:
025 * {@code (0, 1), (1, 3), (2, 1)}. This is encoded as 3 different {@code ref}
026 * variables: {@code ref0={1}, ref1={3}, ref2={1}}.
027 * </p>
028 *
029 * @author jimmy
030 */
031public class PropJoinFunction extends Propagator<Variable> {
032
033    private final SetVar take;
034    private final SetDeltaMonitor takeD;
035    private final IntVar[] refs;
036    private final SetVar to;
037
038    public PropJoinFunction(SetVar take, IntVar[] refs, SetVar to) {
039        super(buildArray(take, to, refs), PropagatorPriority.QUADRATIC, true);
040        this.take = take;
041        this.takeD = take.monitorDelta(aCause);
042        this.refs = refs;
043        this.to = to;
044    }
045
046    private static Variable[] buildArray(SetVar take, SetVar to, IntVar[] refs) {
047        Variable[] array = new Variable[refs.length + 2];
048        array[0] = take;
049        array[1] = to;
050        System.arraycopy(refs, 0, array, 2, refs.length);
051        return array;
052    }
053
054    private boolean isTakeVar(int idx) {
055        return idx == 0;
056    }
057
058    private boolean isToVar(int idx) {
059        return idx == 1;
060    }
061
062    private boolean isRefVar(int idx) {
063        return idx >= 2;
064    }
065
066    private int getRefVarIndex(int idx) {
067        assert isRefVar(idx);
068        return idx - 2;
069    }
070
071    @Override
072    public int getPropagationConditions(int vIdx) {
073        if (isTakeVar(vIdx)) {
074            return EventType.ADD_TO_KER.mask + EventType.REMOVE_FROM_ENVELOPE.mask;
075        }
076        if (isToVar(vIdx)) {
077            return EventType.ADD_TO_KER.mask + EventType.REMOVE_FROM_ENVELOPE.mask;
078        }
079        assert isRefVar(vIdx);
080        return EventType.INT_ALL_MASK();
081    }
082
083    private boolean findMate(int toEnv) throws ContradictionException {
084        boolean inKer = to.kernelContains(toEnv);
085        int mate = -1;
086        for (int j = take.getEnvelopeFirst(); j != SetVar.END; j = take.getEnvelopeNext()) {
087            if (refs[j].contains(toEnv)) {
088                // Found a second mate.
089                if (mate != -1 || !inKer) {
090                    mate = -2;
091                    break;
092                }
093                mate = j;
094            }
095        }
096        if (mate == -1) {
097            // No mates.
098            to.removeFromEnvelope(toEnv, aCause);
099        } else if (mate != -2 && inKer) {
100            // One mate.
101            take.addToKernel(mate, aCause);
102            return refs[mate].instantiateTo(toEnv, aCause);
103        }
104        return false;
105    }
106
107    private void findMates() throws ContradictionException {
108        for (int i = to.getEnvelopeFirst(); i != SetVar.END; i = to.getEnvelopeNext()) {
109            if (findMate(i)) {
110                findMates();
111                return;
112            }
113        }
114    }
115
116    @Override
117    public void propagate(int evtmask) throws ContradictionException {
118        // Pick to and prune refs
119        for (int i = take.getKernelFirst(); i != SetVar.END; i = take.getKernelNext()) {
120            PropUtil.intSubsetEnv(refs[i], to, aCause);
121            if (refs[i].instantiated()) {
122                int value = refs[i].getValue();
123                to.addToKernel(value, aCause);
124            }
125        }
126
127        // Prune take
128        for (int i = take.getEnvelopeFirst(); i != SetVar.END; i = take.getEnvelopeNext()) {
129            if (!PropUtil.domainIntersectEnv(refs[i], to)) {
130                take.removeFromEnvelope(i, aCause);
131            }
132        }
133
134        // Prune to
135        findMates();
136    }
137
138    @Override
139    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
140        if (isTakeVar(idxVarInProp)) {
141            takeD.freeze();
142            takeD.forEach(pruneToOnTakeEnv, EventType.REMOVE_FROM_ENVELOPE);
143            takeD.forEach(pickToAndPruneChildOnTakeKer, EventType.ADD_TO_KER);
144            takeD.unfreeze();
145        } else {
146            forcePropagate(EventType.FULL_PROPAGATION);
147        }
148    }
149    private final IntProcedure pruneToOnTakeEnv = new IntProcedure() {
150        @Override
151        public void execute(int takeEnv) throws ContradictionException {
152            assert !take.envelopeContains(takeEnv);
153
154            IntVar ref = refs[takeEnv];
155            int ub = ref.getUB();
156            for (int i = ref.getLB(); i <= ub; i = ref.nextValue(i)) {
157                if (to.envelopeContains(i)) {
158                    if (findMate(i)) {
159                        findMates();
160                        return;
161                    }
162                }
163            }
164        }
165    };
166    private final IntProcedure pickToAndPruneChildOnTakeKer = new IntProcedure() {
167        @Override
168        public void execute(int takeKer) throws ContradictionException {
169            assert take.kernelContains(takeKer);
170
171            IntVar ref = refs[takeKer];
172            PropUtil.intSubsetEnv(ref, to, aCause);
173            if (ref.instantiated()) {
174                to.addToKernel(ref.getValue(), aCause);
175            }
176        }
177    };
178
179    @Override
180    public ESat isEntailed() {
181        for (int i = take.getKernelFirst(); i != SetVar.END; i = take.getKernelNext()) {
182            if (!PropUtil.domainIntersectEnv(refs[i], to)) {
183                return ESat.FALSE;
184            }
185        }
186        if (!take.instantiated() || !to.instantiated()) {
187            return ESat.UNDEFINED;
188        }
189        FixedCapacityIntSet values = new FixedCapacityIntSet(refs.length);
190        for (IntVar ref : refs) {
191            if (!ref.instantiated()) {
192                return ESat.UNDEFINED;
193            }
194            values.add(ref.getValue());
195        }
196        return values.containsAll(to.getValue()) ? ESat.TRUE : ESat.UNDEFINED;
197    }
198
199    @Override
200    public String toString() {
201        return "joinFunction(" + take + ", " + Arrays.toString(refs) + ", " + to + ")";
202    }
203}
