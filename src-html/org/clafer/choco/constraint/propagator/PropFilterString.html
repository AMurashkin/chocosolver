001package org.clafer.choco.constraint.propagator;
002
003import java.util.Arrays;
004import solver.constraints.propagators.Propagator;
005import solver.constraints.propagators.PropagatorPriority;
006import solver.exception.ContradictionException;
007import solver.variables.EventType;
008import solver.variables.IntVar;
009import solver.variables.SetVar;
010import solver.variables.SetVarImpl;
011import solver.variables.Variable;
012import util.ESat;
013
014/**
015 * result = [string !! i | i <- set].
016 *
017 * Pads result with extra negative one if necessary.
018 *
019 * @author jimmy
020 */
021public class PropFilterString extends Propagator<Variable> {
022
023    private final SetVar set;
024    // Sorted in decreasing order. Non-negatives
025    private final IntVar[] string;
026    private final IntVar[] result;
027
028    public PropFilterString(SetVar set, IntVar[] string, IntVar[] result) {
029        super(buildArray(set, string, result), PropagatorPriority.LINEAR, true);
030        this.set = set;
031        this.string = string;
032        this.result = result;
033    }
034
035    public static Variable[] buildArray(SetVar set, IntVar[] string, IntVar[] result) {
036        Variable[] array = new Variable[1 + string.length + result.length];
037        array[0] = set;
038        System.arraycopy(string, 0, array, 1, string.length);
039        System.arraycopy(result, 0, array, 1 + string.length, result.length);
040        return array;
041    }
042
043    private boolean isSetVar(int idx) {
044        return idx == 0;
045    }
046
047    private boolean isStringVar(int idx) {
048        return idx > 0 && idx <= string.length;
049    }
050
051    private int getStringVarIndex(int idx) {
052        assert isStringVar(idx);
053        return idx - 1;
054    }
055
056    private boolean isResultVar(int idx) {
057        return idx > string.length;
058    }
059
060    private int getResultVarIndex(int idx) {
061        assert isResultVar(idx);
062        return idx - 1 - string.length;
063    }
064
065    @Override
066    public int getPropagationConditions(int vIdx) {
067        if (isSetVar(vIdx)) {
068            return EventType.ADD_TO_KER.mask + EventType.REMOVE_FROM_ENVELOPE.mask;
069        }
070        return EventType.INT_ALL_MASK();
071    }
072
073    private void subset(IntVar a, IntVar b) throws ContradictionException {
074        int ub = a.getUB();
075        for (int i = a.getLB(); i <= ub; i = a.nextValue(i)) {
076            if (!b.contains(i)) {
077                a.removeValue(i, aCause);
078            }
079        }
080    }
081
082    @Override
083    public void propagate(int evtmask) throws ContradictionException {
084        int index = 0;
085        for (int i = set.getEnvelopeFirst(); i != SetVar.END; i = set.getEnvelopeNext()) {
086            if (!set.kernelContains(i)) {
087                return;
088            }
089            subset(string[i], result[index]);
090            subset(result[index], string[i]);
091            index++;
092        }
093        for (; index < result.length; index++) {
094            result[index].instantiateTo(-1, aCause);
095        }
096    }
097
098    @Override
099    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
100        forcePropagate(EventType.FULL_PROPAGATION);
101    }
102
103    @Override
104    public ESat isEntailed() {
105        if (!isCompletelyInstantiated()) {
106            return ESat.UNDEFINED;
107        }
108        int index = 0;
109        for (int i = set.getKernelFirst(); i != SetVar.END; i = set.getKernelNext(), index++) {
110            if (string[i].getValue() != result[index].getValue()) {
111                return ESat.FALSE;
112            }
113        }
114        for (; index < result.length; index++) {
115            if (result[index].getValue() != -1) {
116                return ESat.FALSE;
117            }
118        }
119        return ESat.TRUE;
120    }
121
122    @Override
123    public String toString() {
124        return "filter(" + set + ", " + Arrays.toString(string) + ", " + Arrays.toString(result) + ")";
125    }
126}
