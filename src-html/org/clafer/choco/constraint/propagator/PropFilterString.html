001package org.clafer.choco.constraint.propagator;
002
003import java.util.Arrays;
004import solver.constraints.Propagator;
005import solver.constraints.PropagatorPriority;
006import solver.exception.ContradictionException;
007import solver.variables.EventType;
008import solver.variables.IntVar;
009import solver.variables.SetVar;
010import solver.variables.Variable;
011import util.ESat;
012
013/**
014 * result = [string !! i | i <- set].
015 *
016 * Pads result with extra negative one if necessary.
017 *
018 * @author jimmy
019 */
020public class PropFilterString extends Propagator<Variable> {
021
022    private final SetVar set;
023    private final int offset;
024    // Sorted in decreasing order. Non-negatives
025    private final IntVar[] string;
026    private final IntVar[] result;
027
028    public PropFilterString(SetVar set, int offset, IntVar[] string, IntVar[] result) {
029        super(buildArray(set, string, result), PropagatorPriority.LINEAR, false);
030        this.set = set;
031        this.offset = offset;
032        this.string = string;
033        this.result = result;
034    }
035
036    public static Variable[] buildArray(SetVar set, IntVar[] string, IntVar[] result) {
037        Variable[] array = new Variable[1 + string.length + result.length];
038        array[0] = set;
039        System.arraycopy(string, 0, array, 1, string.length);
040        System.arraycopy(result, 0, array, 1 + string.length, result.length);
041        return array;
042    }
043
044    private boolean isSetVar(int idx) {
045        return idx == 0;
046    }
047
048    private boolean isStringVar(int idx) {
049        return idx > 0 && idx <= string.length;
050    }
051
052    private int getStringVarIndex(int idx) {
053        assert isStringVar(idx);
054        return idx - 1;
055    }
056
057    private boolean isResultVar(int idx) {
058        return idx > string.length;
059    }
060
061    private int getResultVarIndex(int idx) {
062        assert isResultVar(idx);
063        return idx - 1 - string.length;
064    }
065
066    @Override
067    public int getPropagationConditions(int vIdx) {
068        if (isSetVar(vIdx)) {
069            return EventType.ADD_TO_KER.mask + EventType.REMOVE_FROM_ENVELOPE.mask;
070        }
071        return EventType.INT_ALL_MASK();
072    }
073
074    private boolean subset(IntVar sub, IntVar[] sups, int from, int to) throws ContradictionException {
075        boolean changed = false;
076        int ub = sub.getUB();
077        for (int val = sub.getLB(); val <= ub; val = sub.nextValue(val)) {
078            boolean found = false;
079            for (int i = from; i < to; i++) {
080                if (sups[i].contains(val)) {
081                    found = true;
082                    break;
083                }
084            }
085            if (!found) {
086                changed |= sub.removeValue(val, aCause);
087            }
088        }
089        return changed;
090    }
091
092    private boolean subset(IntVar sub, IntVar[] sups, int[] indices, int low, int high) throws ContradictionException {
093        boolean changed = false;
094        int ub = sub.getUB();
095        for (int val = sub.getLB(); val <= ub; val = sub.nextValue(val)) {
096            boolean found = false;
097            for (int i = low; i <= high; i++) {
098                if (sups[indices[i]].contains(val)) {
099                    found = true;
100                    break;
101                }
102            }
103            if (!found) {
104                changed |= sub.removeValue(val, aCause);
105            }
106        }
107        return changed;
108    }
109
110    private boolean subsetOrNegativeOne(IntVar sub, IntVar[] sups, int[] indices, int from, int to) throws ContradictionException {
111        boolean changed = false;
112        int ub = sub.getUB();
113        for (int val = sub.getLB(); val <= ub; val = sub.nextValue(val)) {
114            if (val != -1) {
115                boolean found = false;
116                for (int i = from; i < to; i++) {
117                    assert i < indices.length;
118                    if (sups[indices[i]].contains(val)) {
119                        found = true;
120                        break;
121                    }
122                }
123                if (!found) {
124                    changed |= sub.removeValue(val, aCause);
125                }
126            }
127        }
128        return changed;
129    }
130
131    @Override
132    public void propagate(int evtmask) throws ContradictionException {
133        if (set.getKernelSize() > result.length) {
134            contradiction(set, "Too many in kernel");
135        }
136
137        int[] env = new int[set.getEnvelopeSize()];
138        int[] kerIndices = new int[set.getKernelSize()];
139
140        boolean changed;
141        do {
142            changed = false;
143            // The number of ker elements seen.
144            int kerIndex = 0;
145            // The number of env elements seen.
146            int envIndex = 0;
147            for (int i = set.getEnvelopeFirst(); i != SetVar.END; i = set.getEnvelopeNext()) {
148                int x = i - offset;
149                env[envIndex] = x;
150                if (set.kernelContains(i)) {
151                    kerIndices[kerIndex] = envIndex;
152                    changed |= subset(string[x], result, kerIndex, Math.min(envIndex + 1, result.length));
153                    envIndex++;
154                    kerIndex++;
155                } else {
156                    boolean found = false;
157                    for (int j = kerIndex; j < result.length && j <= envIndex; j++) {
158                        if (PropUtil.domainIntersectDomain(string[x], result[j])) {
159                            found = true;
160                            break;
161                        }
162                    }
163                    if (!found) {
164                        changed |= set.removeFromEnvelope(i, aCause);
165                    } else {
166                        envIndex++;
167                    }
168                }
169            }
170            assert envIndex <= env.length;
171            assert kerIndex == kerIndices.length;
172
173            for (int i = 0; i < result.length; i++) {
174                if (i < kerIndices.length) {
175                    changed |= subset(result[i], string, env, i, kerIndices[i]);
176                } else {
177                    changed |= subsetOrNegativeOne(result[i], string, env, i, envIndex);
178                }
179            }
180        } while (changed);
181
182        if (set.instantiated()) {
183            for (int i = set.getKernelSize(); i < result.length; i++) {
184                result[i].instantiateTo(-1, aCause);
185            }
186        } else {
187            int i = set.getKernelSize();
188            for (; i < result.length && !result[i].contains(-1); i++) {
189            }
190            if (set.getEnvelopeSize() < i) {
191                contradiction(set, "Too few in envelope");
192            }
193        }
194    }
195
196    @Override
197    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
198        forcePropagate(EventType.FULL_PROPAGATION);
199    }
200
201    @Override
202    public ESat isEntailed() {
203        if (!isCompletelyInstantiated()) {
204            return ESat.UNDEFINED;
205        }
206        int index = 0;
207        for (int i = set.getKernelFirst(); i != SetVar.END; i = set.getKernelNext(), index++) {
208            if (index >= result.length) {
209                return ESat.FALSE;
210            }
211            int x = i - offset;
212            if (string[x].getValue() != result[index].getValue()) {
213                return ESat.FALSE;
214            }
215        }
216        for (; index < result.length; index++) {
217            if (result[index].getValue() != -1) {
218                return ESat.FALSE;
219            }
220        }
221        return ESat.TRUE;
222    }
223
224    @Override
225    public String toString() {
226        return "filter(" + set + " >> " + offset + ", " + Arrays.toString(string) + ", " + Arrays.toString(result) + ")";
227    }
228}
