001package org.clafer.choco.constraint.propagator;
002
003import org.clafer.common.Util;
004import solver.constraints.propagators.Propagator;
005import solver.constraints.propagators.PropagatorPriority;
006import solver.exception.ContradictionException;
007import solver.variables.EventType;
008import solver.variables.IntVar;
009import solver.variables.SetVar;
010import solver.variables.Variable;
011import util.ESat;
012
013/**
014 * Sums a set and |set| &le n. This implementation <b>assumes</b> that the
015 * envelope and kernel are sorted, undefined behaviour otherwise.
016 *
017 * @author jimmy
018 */
019public class PropSetSumN extends Propagator<Variable> {
020
021    private final SetVar set;
022    private final IntVar sum;
023    private final int n;
024
025    public PropSetSumN(SetVar set, IntVar sum, int n) {
026        super(new Variable[]{set, sum}, PropagatorPriority.LINEAR, true);
027        if (n <= 0) {
028            throw new IllegalArgumentException();
029        }
030        this.set = set;
031        this.sum = sum;
032        this.n = n;
033    }
034
035    private boolean isSetVar(int idx) {
036        return idx == 0;
037    }
038
039    private boolean isSumVar(int idx) {
040        return idx == 1;
041    }
042
043    @Override
044    public int getPropagationConditions(int vIdx) {
045        if (isSetVar(vIdx)) {
046            return EventType.ADD_TO_KER.mask + EventType.REMOVE_FROM_ENVELOPE.mask;
047        }
048        assert isSumVar(vIdx);
049        return EventType.INSTANTIATE.mask + EventType.BOUND.mask;
050    }
051
052    @Override
053    public void propagate(int evtmask) throws ContradictionException {
054        propagate(0, 0);
055    }
056
057    @Override
058    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
059        int envSize = set.getEnvelopeSize();
060        int kerSize = set.getKernelSize();
061        if (kerSize > n) {
062            contradiction(set, "|" + set + "| > " + n);
063        }
064
065        final int[] ker = PropUtil.iterateKer(set);
066        final int kerSum = Util.sum(ker);
067        // The sum of the lowest n negative integers that can be chosen.
068        int low = kerSum;
069        // The sum of the highest n positive integers that can be chosen.
070        int high = kerSum;
071        // The number of elements seen in env but not in ker.
072        int index = 0;
073        // The size of the set of integers to still choose from.
074        int chooseSize = envSize - kerSize;
075        // The [0-lowEnd) integers in env-ker are the lowest integers.
076        int lowEnd = Math.min(n, envSize) - kerSize;
077        // The [highStart-choosSize) integers in env-ker are the highest integers.
078        int highStart = chooseSize - lowEnd;
079        // The n + 1 highest integer, or 0 if not positive.
080        int highCandidate = 0;
081        // The highest n positive integers in descending order.
082        int[] highChoose = new int[chooseSize - highStart];
083        for (int i = set.getEnvelopeFirst(); i != SetVar.END; i = set.getEnvelopeNext()) {
084            if (!Util.in(i, ker)) {
085                if (index < lowEnd) {
086                    low += Math.min(0, i);
087                }
088                index++;
089                if (index == highStart) {
090                    highCandidate = Math.max(0, i);
091                } else if (index > highStart && i > 0) {
092                    int j = Math.max(0, i);
093                    highChoose[chooseSize - index] = j;
094                    high += j;
095                }
096            }
097        }
098
099        sum.updateLowerBound(low, aCause);
100        sum.updateUpperBound(high, aCause);
101
102        int lb = sum.getLB();
103        int ub = sum.getUB();
104
105        boolean again = false;
106        index = 0;
107        // sum of the lowest n - 1 negative integers that can be chosen.
108        int lowNMinusOne = kerSum;
109        // sum of the highest n - 1 positive integers that can be chosen.
110        int highNMinusOne = high - (highChoose.length == 0 ? 0 : highChoose[highChoose.length - 1]);
111        for (int i = set.getEnvelopeFirst(); i != SetVar.END; i = set.getEnvelopeNext()) {
112            if (!Util.in(i, ker)) {
113                if (index < lowEnd - 1) {
114                    lowNMinusOne += Math.min(0, i);
115                } else if ( // Is it possible to add i to the kernel?
116                        // Adding i will never be under the upper bound
117                        lowNMinusOne + i > ub
118                        // Adding i will never be over the lower bound
119                        || highNMinusOne + i < lb) {
120                    // With i, it is impossible to stay below the upper bound.
121                    // Remove it from the envelope.
122                    again |= set.removeFromEnvelope(i, aCause);
123                }
124                index++;
125            }
126        }
127        for (int i = highChoose.length - 1; i >= 0
128                // Is it possible to leave i out of the kernel?
129                // Replace i with the candidate.
130                // Not adding highChoose[i] will never be over the lower bound
131                && high - highChoose[i] + highCandidate < lb;
132                i--) {
133            // Without i, it is impossible to reach the lower bound.
134            // Add it to the kernel.
135            again |= set.addToKernel(highChoose[i], aCause);
136        }
137        if (again) {
138            // Changed some values, need to repropagate.
139            propagate(0, 0);
140        }
141    }
142
143    @Override
144    public ESat isEntailed() {
145        int envSize = set.getEnvelopeSize();
146        int kerSize = set.getKernelSize();
147        if (kerSize > n) {
148            return ESat.FALSE;
149        }
150        int[] ker = PropUtil.iterateKer(set);
151        int low = 0;
152        int high = 0;
153        // The number of elements seen in env but not in ker.
154        int index = 0;
155        int lowEnd = Math.min(n, envSize - kerSize);
156        int highStart = envSize - kerSize - lowEnd;
157        for (int i = set.getEnvelopeFirst(); i != SetVar.END; i = set.getEnvelopeNext()) {
158            if (Util.in(i, ker)) {
159                low += i;
160                high += i;
161            } else {
162                if (index < lowEnd && i < 0) {
163                    low += i;
164                }
165                if (index >= highStart && i > 0) {
166                    high += i;
167                }
168                index++;
169            }
170        }
171        if (sum.getLB() > low || sum.getUB() < high) {
172            return ESat.FALSE;
173        }
174        return low == high && sum.instantiated() ? ESat.TRUE : ESat.UNDEFINED;
175    }
176
177    @Override
178    public String toString() {
179        return "sum(" + set + ") = " + sum + " where |" + set.getName() + "| <= " + n;
180    }
181}
