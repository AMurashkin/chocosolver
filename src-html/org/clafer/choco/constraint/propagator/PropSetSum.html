033    private boolean isSetVar(int idx) {
034        return idx == 0;
035    }
036
037    private boolean isSumVar(int idx) {
038        return idx == 1;
039    }
040
041    private boolean isCardVar(int idx) {
042        return idx == 2;
043    }

045    @Override
046    public int getPropagationConditions(int vIdx) {
047        if (isSetVar(vIdx)) {
048            return EventType.ADD_TO_KER.mask + EventType.REMOVE_FROM_ENVELOPE.mask;
049        }
050        assert isSumVar(vIdx) || isCardVar(vIdx);
051        return EventType.INSTANTIATE.mask + EventType.BOUND.mask;
052    }

054    private Triple<int[], int[], int[]> findKerSmallestLargest(int n) {
055        final int[] ker = PropUtil.iterateKer(set);
056        final int[] smallest = new int[n];
057        final int[] largest = new int[n];
058
059        int chooseSize = set.getEnvelopeSize() - ker.length;
060        int highStart = chooseSize - n;
061
062        int index = 0;
063        for (int i = set.getEnvelopeFirst(); i != SetVar.END; i = set.getEnvelopeNext()) {
064            if (!Util.in(i, ker)) {
065                if (index < smallest.length) {
066                    smallest[index] = i;
067                }
068                index++;
069                if (index > highStart) {
070                    largest[chooseSize - index] = i;
071                }
072            }
073        }
074        return new Triple<int[], int[], int[]>(ker, smallest, largest);
075    }
076
077    private int lowestMinus(int val, int[] ker, int[] free) {
078        int lb = card.getLB() - ker.length;
079        int ub = card.getUB() - ker.length;
080        int index = 0;
081        int low = Util.sum(ker);
082        for (int i = 0; i < free.length; i++) {
083            if (free[i] != val) {
084                if (index < lb) {
085                    low += free[i];
086                } else if (index < ub) {
087                    low += Math.min(free[i], 0);
088                } else {
089                    break;
090                }
091                index++;
092            }
093        }
094        return low;
095    }

270    @Override
271    public ESat isEntailed() {
272        int envSize = set.getEnvelopeSize();
273        int kerSize = set.getKernelSize();
274        if (kerSize > card.getUB()) {
275            return ESat.FALSE;
276        }
277        int[] ker = PropUtil.iterateKer(set);
278        int low = 0;
279        int high = 0;
280        // The number of elements seen in env but not in ker.
281        int index = 0;
282        int lowEnd = Math.min(card.getLB(), envSize) - kerSize;
283        int highStart = envSize - kerSize - lowEnd;
284        for (int i = set.getEnvelopeFirst(); i != SetVar.END; i = set.getEnvelopeNext()) {
285            if (Util.in(i, ker)) {
286                low += i;
287                high += i;
288            } else {
289                if (index < lowEnd && i < 0) {
290                    low += i;
291                }
292                if (index >= highStart && i > 0) {
293                    high += i;
294                }
295                index++;
296            }
297        }
298        if (sum.getLB() > low || sum.getUB() < high) {
299            return ESat.FALSE;
300        }
301        return low == high && sum.instantiated() ? ESat.TRUE : ESat.UNDEFINED;
302    }
