028    // Must be non-negative.
029    // SCALE MUST BE LARGER THAN ANY ONE WEIGHT
030    private final int scale;
031    // Must be non-negative.
032    private final IntVar sum;
033
034    public PropSumWeight(SetVar set, IntVar[] weight, int scale, IntVar sum) {
035        super(buildArray(set, weight, sum), PropagatorPriority.LINEAR, true);
036        this.set = set;
037        this.weight = weight;
038        this.scale = scale;
039        this.sum = sum;
040    }
041
042    public static Variable[] buildArray(SetVar set, IntVar[] weight, IntVar sum) {
043        Variable[] array = new Variable[weight.length + 2];
044        array[0] = set;
045        array[1] = sum;
046        System.arraycopy(weight, 0, array, 2, weight.length);
047        return array;
048    }
049
050    private boolean isSetVar(int idx) {
051        return idx == 0;
052    }
053
054    private boolean isSumVar(int idx) {
055        return idx == 1;
056    }
057
058    private boolean isWeightVar(int idx) {
059        return idx >= 2;
060    }
061
062    private int getWeightVarIndex(int idx) {
063        assert isWeightVar(idx);
064        return idx - 2;
065    }
066
067    @Override

072        return EventType.INT_ALL_MASK();

118    @Override
119    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
120        forcePropagate(EventType.FULL_PROPAGATION);
121    }
122
123    @Override
124    public ESat isEntailed() {
125        int minSum = 0;
126        for (int i = set.getKernelFirst(); i != SetVar.END; i = set.getKernelNext()) {
127            minSum = minSum * scale + weight[i].getLB();
128        }
129        if (minSum > sum.getUB()) {
130            return ESat.FALSE;
131        }
132        int maxSum = 0;
133        for (int i = set.getEnvelopeFirst(); i != SetVar.END; i = set.getEnvelopeNext()) {
134            maxSum = maxSum * scale + weight[i].getUB();
135        }
136        if (maxSum < sum.getLB()) {
137            return ESat.FALSE;
138        }
139        return minSum == maxSum && sum.instantiated() ? ESat.TRUE : ESat.UNDEFINED;
140    }
