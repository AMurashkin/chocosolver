001package org.clafer.compiler;
002
003import java.util.HashSet;
004import java.util.Set;
005import org.clafer.ast.AstConstraint;
006import org.clafer.common.Check;
007import org.clafer.collection.Pair;
008import org.clafer.instance.InstanceModel;
009import solver.ResolutionPolicy;
010import solver.Solver;
011import solver.variables.BoolVar;
012import solver.variables.IntVar;
013import util.ESat;
014
015/**
016 *
017 * @author jimmy
018 */
019public class ClaferUnsat {
020
021    private final Solver solver;
022    private final ClaferSolutionMap solutionMap;
023    private final Pair<AstConstraint, BoolVar>[] softVars;
024    private final IntVar score;
025
026    ClaferUnsat(Solver solver, ClaferSolutionMap solutionMap, Pair<AstConstraint, BoolVar>[] softVars, IntVar score) {
027        this.solver = Check.notNull(solver);
028        this.solutionMap = Check.notNull(solutionMap);
029        this.softVars = Check.noNulls(softVars);
030        this.score = Check.notNull(score);
031    }
032
033    public Solver getInternalSolver() {
034        return solver;
035    }
036
037    public Pair<Set<AstConstraint>, InstanceModel> minUnsat() {
038        solver.findOptimalSolution(ResolutionPolicy.MAXIMIZE, score);
039        if (ESat.TRUE.equals(solver.isFeasible())) {
040            Set<AstConstraint> unsat = new HashSet<AstConstraint>();
041            for (Pair<AstConstraint, BoolVar> softVar : softVars) {
042                if (softVar.getSnd().instantiatedTo(0)) {
043                    unsat.add(softVar.getFst());
044                }
045            }
046            return new Pair<Set<AstConstraint>, InstanceModel>(unsat, solutionMap.getInstance());
047        }
048        return null;
049    }
050}
