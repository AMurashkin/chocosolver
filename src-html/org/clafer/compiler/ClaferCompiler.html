040    public static ClaferSolver compile(AstModel in, ScopeBuilder scope) {
041        return compile(in, scope.toScope());
042    }
043
044    public static ClaferSolver compile(AstModel in, Scope scope) {
045        Solver solver = new Solver();
046        IrModule module = new IrModule();
047
048        AstSolutionMap astSolution = AstCompiler.compile(in, scope, module);
049        IrSolutionMap irSolution = IrCompiler.compile(module, solver);
050        ClaferSolutionMap solution = new ClaferSolutionMap(astSolution, irSolution);
051
052        solver.set(new StrategiesSequencer(solver.getEnvironment(),
053                SetStrategyFactory.setLex(solution.getIrSolution().getSetVars()),
054                IntStrategyFactory.firstFail_InDomainMin(solution.getIrSolution().getIntVars()),
055                IntStrategyFactory.firstFail_InDomainMin(solution.getIrSolution().getBoolVars())));
056        return new ClaferSolver(solver, solution);
057    }
058
059    public static ClaferObjective compileMaximize(AstModel in, ScopeBuilder scope, AstRef ref) {
060        return compileMaximize(in, scope.toScope(), ref);
061    }
062
063    public static ClaferObjective compileMaximize(AstModel in, Scope scope, AstRef ref) {
064        Solver solver = new Solver();
065        IrModule module = new IrModule();
066
067        AstSolutionMap astSolution = AstCompiler.compile(in, scope, module);
068        IrSolutionMap irSolution = IrCompiler.compile(module, solver);
069        ClaferSolutionMap solution = new ClaferSolutionMap(astSolution, irSolution);
070
071        IntVar[] score = irSolution.getIntVars(astSolution.getRefVars(ref));
072        int[] bounds = Sum.getSumBounds(score);
073        IntVar sum = VF.bounded("Score", bounds[0], bounds[1], solver);
074        solver.post(ICF.sum(score, sum));
075
076        solver.set(new StrategiesSequencer(solver.getEnvironment(),
077                SetStrategyFactory.setLex(solution.getIrSolution().getSetVars()),
078                IntStrategyFactory.firstFail_InDomainMax(score),
079                IntStrategyFactory.firstFail_InDomainMax(solution.getIrSolution().getIntVars()),
080                IntStrategyFactory.firstFail_InDomainMax(solution.getIrSolution().getBoolVars())));
081        return new ClaferObjective(solver, solution, Objective.Maximize, sum);
082    }

109    public static ClaferUnsat compileUnsat(AstModel in, ScopeBuilder scope) {
110        return compileUnsat(in, scope.toScope());
111    }
112
113    public static ClaferUnsat compileUnsat(AstModel in, Scope scope) {
114        Solver solver = new Solver();
115        IrModule module = new IrModule();
116
117        AstSolutionMap astSolution = AstCompiler.compile(in, scope, module,
118                Util.cons(new UnsatAnalyzer(), AstCompiler.DefaultAnalyzers));
119        IrSolutionMap irSolution = IrCompiler.compile(module, solver);
120        ClaferSolutionMap solution = new ClaferSolutionMap(astSolution, irSolution);
121
122        Pair<AstConstraint, IrBoolVar>[] irSoftVarPairs = astSolution.getSoftVars();
123        BoolVar[] softVars = new BoolVar[irSoftVarPairs.length];
124        @SuppressWarnings("unchecked")
125        Pair<AstConstraint, BoolVar>[] softVarPairs = new Pair[irSoftVarPairs.length];
126        for(int i = 0; i < softVars.length;i++) {
127            softVars[i] = irSolution.getBoolVar(irSoftVarPairs[i].getSnd());
128            softVarPairs[i] = new Pair<AstConstraint, BoolVar>(irSoftVarPairs[i].getFst(), softVars[i]);
129        }
130        int[] bounds = Sum.getSumBounds(softVars);
131        IntVar sum = VF.bounded("Score", bounds[0], bounds[1], solver);
132        solver.post(ICF.sum(softVars, sum));
133
134        solver.set(new StrategiesSequencer(solver.getEnvironment(),
135                IntStrategyFactory.firstFail_InDomainMax(softVars),
136                IntStrategyFactory.firstFail_InDomainMin(solution.getIrSolution().getIntVars()),
137                SetStrategyFactory.setLex(solution.getIrSolution().getSetVars()),
138                IntStrategyFactory.firstFail_InDomainMin(solution.getIrSolution().getBoolVars())));
139        
140        return new ClaferUnsat(solver, solution, softVarPairs, sum);
141    }
