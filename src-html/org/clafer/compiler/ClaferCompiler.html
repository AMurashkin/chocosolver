060                if (!(setVar instanceof IrSetConstant)) {

076    private static IntVar[] getIntVars(AstModel model, ClaferSolutionMap map) {
077        List<IntVar> vars = new ArrayList<IntVar>();
078        for (AstClafer clafer : AstUtil.getClafers(model)) {
079            if (clafer.hasRef()) {
080                for (IrIntVar intVar : map.getAstSolution().getRefVars(clafer.getRef())) {
081                    if (!(intVar instanceof IrIntConstant)) {
082                        vars.add(map.getIrSolution().getIntVar(intVar));
083                    }
084                }
085            }
086        }
087        return vars.toArray(new IntVar[vars.size()]);
088    }

090    public static ClaferSolver compile(AstModel in, ScopeBuilder scope) {
091        return compile(in, scope.toScope());
092    }

094    public static ClaferSolver compile(AstModel in, Scope scope) {
095        Solver solver = new Solver();
096        IrModule module = new IrModule();
097
098        AstSolutionMap astSolution = AstCompiler.compile(in, scope, module);
099        IrSolutionMap irSolution = IrCompiler.compile(module, solver);
100        ClaferSolutionMap solution = new ClaferSolutionMap(astSolution, irSolution);
101
102        solver.set(new StrategiesSequencer(solver.getEnvironment(),
103                SetStrategyFactory.setLex(getSetVars(in, solution)),
104                IntStrategyFactory.firstFail_InDomainMin(getIntVars(in, solution)),
105                IntStrategyFactory.firstFail_InDomainMax(solution.getIrSolution().getBoolDecisionVars())));
106        return new ClaferSolver(solver, solution);
107    }

113    public static ClaferObjective compileMaximize(AstModel in, Scope scope, AstRef ref) {
114        Solver solver = new Solver();
115        IrModule module = new IrModule();
116
117        Triple<AstSolutionMap, IrIntVar[], IrIntVar> triple = AstCompiler.compile(in, scope, ref, module);
118        AstSolutionMap astSolution = triple.getFst();
119        IrSolutionMap irSolution = IrCompiler.compile(module, solver);
120        ClaferSolutionMap solution = new ClaferSolutionMap(astSolution, irSolution);
121
122        solver.set(new StrategiesSequencer(solver.getEnvironment(),
123                SetStrategyFactory.setLex(getSetVars(in, solution)),
124                IntStrategyFactory.firstFail_InDomainMax(irSolution.getIntVars(triple.getSnd())),
125                IntStrategyFactory.firstFail_InDomainMin(getIntVars(in, solution)),
126                IntStrategyFactory.firstFail_InDomainMax(irSolution.getBoolDecisionVars())));
127        return new ClaferObjective(solver, solution, Objective.Maximize, irSolution.getIntVar(triple.getThd()));
128    }

155    public static ClaferUnsat compileUnsat(AstModel in, Scope scope) {
156        Solver solver = new Solver();
157        IrModule module = new IrModule();
158
159        AstSolutionMap astSolution = AstCompiler.compile(in, scope, module,
160                Util.cons(new UnsatAnalyzer(), AstCompiler.DefaultAnalyzers));
161        IrSolutionMap irSolution = IrCompiler.compile(module, solver);
162        ClaferSolutionMap solution = new ClaferSolutionMap(astSolution, irSolution);
163
164        Pair<AstConstraint, IrBoolVar>[] irSoftVarPairs = astSolution.getSoftVars();
165        BoolVar[] softVars = new BoolVar[irSoftVarPairs.length];
166        @SuppressWarnings("unchecked")
167        Pair<AstConstraint, BoolVar>[] softVarPairs = new Pair[irSoftVarPairs.length];
168        for (int i = 0; i < softVars.length; i++) {
169            softVars[i] = irSolution.getBoolVar(irSoftVarPairs[i].getSnd());
170            softVarPairs[i] = new Pair<AstConstraint, BoolVar>(irSoftVarPairs[i].getFst(), softVars[i]);
171        }
172        int[] bounds = Sum.getSumBounds(softVars);
173        IntVar sum = VF.bounded("Score", bounds[0], bounds[1], solver);
174        solver.post(ICF.sum(softVars, sum));
175
176        solver.set(new StrategiesSequencer(solver.getEnvironment(),
177                IntStrategyFactory.firstFail_InDomainMax(softVars),
178                SetStrategyFactory.setLex(getSetVars(in, solution)),
179                IntStrategyFactory.firstFail_InDomainMin(getIntVars(in, solution)),
180                IntStrategyFactory.firstFail_InDomainMax(solution.getIrSolution().getBoolDecisionVars())));
181        return new ClaferUnsat(solver, solution, softVarPairs, sum);
182    }

184    public static ClaferSolver compilePartial(AstModel in, ScopeBuilder scope, AstConcreteClafer... concretize) {
185        return compilePartial(in, scope.toScope(), concretize);
186    }

188    public static ClaferSolver compilePartial(AstModel in, Scope scope, AstConcreteClafer... concretize) {
189        final Set<AstConcreteClafer> transitiveConcretize = new HashSet<AstConcreteClafer>();
190        for (AstConcreteClafer clafer : concretize) {
191            concretize(clafer, transitiveConcretize);
192        }
193        final ClaferSolver solver = compile(in, scope);
194        final List<IntVar> intVars = new ArrayList<IntVar>();
195        final List<SetVar> setVars = new ArrayList<SetVar>();
196        for (AstConcreteClafer clafer : transitiveConcretize) {
197            IrSetVar[] siblingVars = solver.getSolutionMap().getAstSolution().getSiblingVars(clafer);
198            for (IrSetVar var : siblingVars) {
199                if (!(var instanceof IrSetConstant)) {
200                    setVars.add(solver.getSolutionMap().getIrSolution().getSetVar(var));
201                }
202            }
203            AstRef ref = AstUtil.getInheritedRef(clafer);
204            if (ref != null) {
205                IrIntVar[] refVars = solver.getSolutionMap().getAstSolution().getRefVars(ref);
206                for (IrIntVar var : refVars) {
207                    if (!(var instanceof IrIntConstant)) {
208                        intVars.add(solver.getSolutionMap().getIrSolution().getIntVar(var));
209                    }
210                }
211            }
212        }
213        solver.getInternalSolver().getSearchLoop().plugSearchMonitor(new IMonitorSolution() {
214            @Override
215            public void onSolution() {
216                List<Constraint> constraints = new ArrayList<Constraint>();
217                for (IntVar var : intVars) {
218                    constraints.add(ICF.arithm(var, "!=", var.getValue()));
219                }
220                for (SetVar var : setVars) {
221                    constraints.add(Constraints.notEqual(var, var.getValue()));
222                }
223                solver.getInternalSolver().postCut(Constraints.or(constraints));
224            }
225        });
226        return solver;
227    }
228
229    private static void concretize(AstClafer clafer, Set<AstConcreteClafer> concretize) {
230        if (clafer instanceof AstAbstractClafer) {
231            concretize((AstAbstractClafer) clafer, concretize);
232        } else {
233            concretize((AstConcreteClafer) clafer, concretize);
234        }
235    }
236
237    private static void concretize(AstConcreteClafer clafer, Set<AstConcreteClafer> concretize) {
238        if (!AstUtil.isRoot(clafer) && concretize.add(clafer)) {
239            concretize(clafer.getParent(), concretize);
240            concretize(clafer.getSuperClafer(), concretize);
241        }
242    }
243
244    private static void concretize(AstAbstractClafer clafer, Set<AstConcreteClafer> concretize) {
245        if (!AstUtil.isTypeRoot(clafer)) {
246            for (AstClafer sub : clafer.getSubs()) {
247                concretize(sub, concretize);
248            }
249        }
250    }
