001package org.clafer.collection;
002
003/**
004 * @param <K> the type of keys
005 * @param <V> the type of values
006 * @author jimmy
007 */
008public class ReadWriteHashMap<K, V> extends ReadHashMap<K, V> {
009
010    /**
011     * Bind the key to the value.
012     *
013     * @param key the key
014     * @param value the value
015     * @throws IllegalStateException if the key already has a binding
016     */
017    public void put(K key, V value) throws IllegalStateException {
018        if (map.put(key, value) != null) {
019            throw new IllegalStateException(key + " already contained in the map.");
020        }
021    }
022}
