001package org.clafer.javascript;
002
003import org.clafer.scope.Scope;
004import java.util.HashMap;
005import java.util.List;
006import java.util.Map;
007import java.util.Map.Entry;
008import org.clafer.ast.Asts;
009import org.clafer.ast.AstClafer;
010import org.clafer.ast.AstModel;
011import org.clafer.ast.AstUtil;
012
013/**
014 *
015 * @author jimmy
016 */
017public class JavascriptContext {
018
019    private final Map<String, Integer> scope = new HashMap<String, Integer>();
020    private int defaultScope = 1;
021    private int intLow = -16;
022    private int intHigh = 16;
023    private final AstModel model = Asts.newModel();
024
025    public void setScope(Map<String, Double> scope) {
026        for (Entry<String, Double> entry : scope.entrySet()) {
027            // Javascript integers are doubles.
028            this.scope.put(entry.getKey(), entry.getValue().intValue());
029        }
030    }
031
032    public void setDefaultScope(int defaultScope) {
033        if (defaultScope < 1) {
034            throw new IllegalArgumentException();
035        }
036        this.defaultScope = defaultScope;
037    }
038
039    public void setIntRange(int intLow, int intHigh) {
040        if (intLow > intHigh) {
041            throw new IllegalArgumentException();
042        }
043        this.intLow = intLow;
044        this.intHigh = intHigh;
045    }
046
047    public Scope getScope() {
048        Map<AstClafer, Integer> resolvedScope = new HashMap<AstClafer, Integer>();
049        Map<String, AstClafer> resolvedClafers = getClafersMap(model);
050        for (Entry<String, Integer> entry : scope.entrySet()) {
051            String key = entry.getKey();
052            AstClafer clafer = resolvedClafers.get(key);
053            if (clafer == null) {
054                throw new IllegalStateException("Cannot set scope for unknown Clafer \"" + key + "\", "
055                        + "known Clafers " + resolvedClafers.keySet() + ".");
056            }
057            resolvedScope.put(clafer, entry.getValue());
058        }
059        return new Scope(resolvedScope, defaultScope, intLow, intHigh);
060    }
061
062    public AstModel getModel() {
063        return model;
064    }
065
066    private static Map<String, AstClafer> getClafersMap(AstModel model) {
067        List<AstClafer> clafers = AstUtil.getClafers(model);
068        Map<String, AstClafer> map = new HashMap<String, AstClafer>();
069        for (AstClafer clafer : clafers) {
070            map.put(clafer.getName(), clafer);
071        }
072        assert map.size() == clafers.size() : "Duplicate Clafer name";
073        return map;
074    }
075}
