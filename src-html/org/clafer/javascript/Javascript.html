001package org.clafer.javascript;
002
003import java.io.File;
004import java.io.FileNotFoundException;
005import java.io.FileReader;
006import java.io.InputStreamReader;
007import java.io.Reader;
008import javax.script.ScriptEngine;
009import javax.script.ScriptEngineManager;
010import javax.script.ScriptException;
011import org.clafer.ast.AstModel;
012import org.clafer.scope.Scope;
013import org.clafer.collection.Pair;
014
015/**
016 * This class provides various methods of loading models and scopes via the
017 * Javascript API.
018 *
019 * @author jimmy
020 */
021public class Javascript {
022
023    private Javascript() {
024    }
025
026    /**
027     * @return a new Javascript engine.
028     */
029    public static ScriptEngine newEngine() {
030        // The Rhino engine is currently the only supported engine.
031        ScriptEngine engine = new ScriptEngineManager().getEngineByName("rhino");
032        if (engine == null) {
033            throw new IllegalStateException("Missing Rhino Javascript engine.");
034        }
035        return engine;
036    }
037
038    public static Pair<AstModel, Scope> readModel(String in) throws ScriptException {
039        return readModel("<unknown>", in, newEngine());
040    }
041
042    public static Pair<AstModel, Scope> readModel(File in) throws FileNotFoundException, ScriptException {
043        return readModel(in, newEngine());
044    }
045
046    public static Pair<AstModel, Scope> readModel(Reader in) throws ScriptException {
047        return readModel("<unknown>", in, newEngine());
048    }
049
050    public static Pair<AstModel, Scope> readModel(File in, ScriptEngine engine) throws FileNotFoundException, ScriptException {
051        return readModel(in.getName(), in, engine);
052    }
053
054    public static Pair<AstModel, Scope> readModel(String name, String in, ScriptEngine engine) throws ScriptException {
055        JavascriptContext context = new JavascriptContext();
056        engine.put("rc", context);
057        engine.put(ScriptEngine.FILENAME, "header.js");
058        engine.eval(new InputStreamReader(Javascript.class.getResourceAsStream("header.js")));
059        engine.put(ScriptEngine.FILENAME, name);
060        engine.eval(in);
061        return new Pair<AstModel, Scope>(context.getModel(), context.getScope());
062    }
063
064    public static Pair<AstModel, Scope> readModel(String name, File in, ScriptEngine engine) throws FileNotFoundException, ScriptException {
065        return readModel(name, new FileReader(in), engine);
066    }
067
068    public static Pair<AstModel, Scope> readModel(String name, Reader in, ScriptEngine engine) throws ScriptException {
069        JavascriptContext context = new JavascriptContext();
070        engine.put("rc", context);
071        engine.put(ScriptEngine.FILENAME, "header.js");
072        engine.eval(new InputStreamReader(Javascript.class.getResourceAsStream("header.js")));
073        engine.put(ScriptEngine.FILENAME, name);
074        engine.eval(in);
075        return new Pair<AstModel, Scope>(context.getModel(), context.getScope());
076    }
077}
