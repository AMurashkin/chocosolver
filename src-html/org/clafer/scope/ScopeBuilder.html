001package org.clafer.scope;
002
003import java.util.HashMap;
004import java.util.Map;
005import org.clafer.ast.AstClafer;
006
007/**
008 * Builder pattern for scopes. Use {@link Scope#builder()},
009 * {@link Scope#set(org.clafer.ast.AstClafer, int)}, {@link Scope#defaultScope(int)},
010 * {@link Scope#intLow}, or {@link Scope#intHigh} to construct the builder. The
011 * default scope and lowest and highest integers will be given a default if not
012 * set explicitly.
013 * 
014 * @author jimmy
015 * @see Scope
016 */
017public class ScopeBuilder {
018
019    private final Map<AstClafer, Integer> scope = new HashMap<AstClafer, Integer>();
020    private int defaultScope = 1;
021    private int intLow = -16;
022    private int intHigh = 16;
023
024    ScopeBuilder() {
025    }
026
027    /**
028     * Set the scope of the Clafer. If the scope is already set for the Clafer,
029     * then the new scope overrides the previous one.
030     * 
031     * @param clafer the Clafer
032     * @param scope the scope of the clafer
033     * @return this builder
034     */
035    public ScopeBuilder set(AstClafer clafer, int scope) {
036        this.scope.put(clafer, scope);
037        return this;
038    }
039
040    /**
041     * Set the scope for unspecified Clafers. If the default scope is already set,
042     * then the new default scope overrides the previous one.
043     * 
044     * @param defaultScope the default scope
045     * @return this builder
046     */
047    public ScopeBuilder defaultScope(int defaultScope) {
048        this.defaultScope = defaultScope;
049        return this;
050    }
051
052    /**
053     * Set the lowest (inclusive) integer used for solving. If the lowest integer
054     * is already set, then the new lowest integer overrides the previous one.
055     * 
056     * @param intLow the lowest integer
057     * @return this builder
058     */
059    public ScopeBuilder intLow(int intLow) {
060        this.intLow = intLow;
061        return this;
062    }
063
064    /**
065     * Set the highest (inclusive) integer used for solving. If the highest integer
066     * is already set, then the new highest integer overrides the previous one.
067     * 
068     * @param intHigh the highest integer
069     * @return this builder
070     */
071    public ScopeBuilder intHigh(int intHigh) {
072        this.intHigh = intHigh;
073        return this;
074    }
075
076    /**
077     * Finalizes all the decisions made in the builder. Further changes to this
078     * builder is permitted for building more scopes, but the returned scope
079     * will not be affected.
080     * 
081     * @return the built scope
082     */
083    public Scope toScope() {
084        return new Scope(scope, defaultScope, intLow, intHigh);
085    }
086}
