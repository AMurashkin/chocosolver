001package org.clafer.ast;
002
003/**
004 * Do NOT reuse for different expressions.
005 * 
006 * @author jimmy
007 */
008public class AstThis implements AstSetExpr {
009
010    @Override
011    public <A, B> B accept(AstExprVisitor<A, B> visitor, A a) {
012        return visitor.visit(this, a);
013    }
014
015    @Override
016    public String toString() {
017        return "this";
018    }
019}
