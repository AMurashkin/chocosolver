001package org.clafer.ast.analysis;
002
003import java.util.HashSet;
004import java.util.Set;
005import org.clafer.ast.AstClafer;
006import org.clafer.ast.AstConcreteClafer;
007import org.clafer.ast.AstRef;
008import org.clafer.graph.GraphUtil;
009import org.clafer.graph.KeyGraph;
010
011/**
012 *
013 * @author jimmy
014 */
015public class SymmetryAnalyzer implements Analyzer {
016
017    @Override
018    public Analysis analyze(Analysis analysis) {
019        KeyGraph<AstClafer> graph = new KeyGraph<AstClafer>();
020        for (AstClafer clafer : analysis.getClafers()) {
021            if (clafer instanceof AstConcreteClafer) {
022                AstConcreteClafer concreteClafer = (AstConcreteClafer) clafer;
023                if (concreteClafer.hasParent() && analysis.getScope(concreteClafer.getParent()) > 1) {
024                    graph.getVertex(concreteClafer).addNeighbour(graph.getVertex(concreteClafer.getParent()));
025                }
026            }
027            if (analysis.getScope(clafer) > 1) {
028                for (AstConcreteClafer child : clafer.getChildren()) {
029                    if (analysis.getScope(child) > 1) {
030                        graph.getVertex(clafer).addNeighbour(graph.getVertex(child));
031                    }
032                }
033            }
034            if (clafer.hasRef()) {
035                graph.getVertex(clafer).addNeighbour(graph.getVertex(clafer.getRef().getTargetType()));
036            }
037        }
038
039        Set<AstRef> breakableRefs = new HashSet<AstRef>();
040        for (AstClafer clafer : analysis.getClafers()) {
041            if (clafer.hasRef()) {
042                if (!GraphUtil.hasPath(
043                        graph.getVertex(clafer.getRef().getTargetType()),
044                        graph.getVertex(clafer),
045                        graph)) {
046                    breakableRefs.add(clafer.getRef());
047                }
048            }
049        }
050
051        return analysis.withBreakableRefs(breakableRefs);
052    }
053}
