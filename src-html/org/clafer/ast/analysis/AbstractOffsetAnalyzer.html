001package org.clafer.ast.analysis;
002
003import java.util.ArrayList;
004import java.util.HashMap;
005import java.util.List;
006import java.util.Map;
007import org.clafer.ast.AstAbstractClafer;
008import org.clafer.ast.AstClafer;
009
010/**
011 *
012 * @author jimmy
013 */
014public class AbstractOffsetAnalyzer implements Analyzer {
015
016    @Override
017    public Analysis analyze(Analysis analysis) {
018        Map<AstAbstractClafer, Offsets> offsetsMap = new HashMap<AstAbstractClafer, Offsets>();
019        for (AstAbstractClafer abstractClafer : analysis.getAbstractClafers()) {
020            Map<AstClafer, Integer> offsets = new HashMap<AstClafer, Integer>();
021            List<AstClafer> reverseOffsets = new ArrayList<AstClafer>();
022            int offset = 0;
023            List<AstClafer> subs = new ArrayList<AstClafer>(abstractClafer.getSubs());
024            /*
025             * What is this optimization?
026             *
027             * This optimization is to put more "stable" Clafers near the
028             * beginning, to reduce the number of backtracking for LowGroups.
029             */
030            AnalysisUtil.descendingGlobalCardRatio(subs, analysis.getGlobalCardMap());
031            List<AstClafer> greedy = subs;
032            for (AstClafer sub : greedy) {
033                offsets.put(sub, offset);
034                int skip = analysis.getGlobalCard(sub).getHigh();
035                for (int i = 0; i < skip; i++) {
036                    reverseOffsets.add(sub);
037                }
038                offset += skip;
039            }
040            offsetsMap.put(abstractClafer, new Offsets(abstractClafer, offsets, reverseOffsets.toArray(new AstClafer[reverseOffsets.size()])));
041        }
042        return analysis.withOffsetMap(offsetsMap);
043    }
044}
