001package org.clafer.ast.analysis;
002
003import java.util.Collections;
004import java.util.Comparator;
005import java.util.List;
006import java.util.Map;
007import org.clafer.common.Check;
008import org.clafer.ast.AstAbstractClafer;
009import org.clafer.ast.AstClafer;
010import org.clafer.ast.Card;
011
012/**
013 *
014 * @author jimmy
015 */
016@Deprecated
017public class AnalysisUtil {
018
019    private AnalysisUtil() {
020    }
021
022    public static <T> T notNull(String message, T t) {
023        if (t == null) {
024            throw new AnalysisException(message);
025        }
026        return t;
027    }
028
029    public static void descendingDepths(
030            List<AstAbstractClafer> abstractClafers,
031            Map<AstAbstractClafer, Integer> depths) {
032        Collections.sort(abstractClafers, new DepthComparator(depths));
033    }
034
035    public static void descendingGlobalCardRatio(
036            List<AstClafer> clafers,
037            Map<AstClafer, Card> globalCards) {
038        Collections.sort(clafers, new GlobalCardRatioComparator(globalCards));
039    }
040
041    private static class DepthComparator implements Comparator<AstAbstractClafer> {
042
043        private final Map<AstAbstractClafer, Integer> depths;
044
045        DepthComparator(Map<AstAbstractClafer, Integer> depths) {
046            this.depths = Check.notNull(depths);
047        }
048
049        @Override
050        public int compare(AstAbstractClafer o1, AstAbstractClafer o2) {
051            int depth1 = notNull(o1 + " Depth not analyzed yet", depths.get(o1)).intValue();
052            int depth2 = notNull(o2 + " Depth not analyzed yet", depths.get(o2)).intValue();
053            return depth1 > depth2 ? -1 : (depth1 == depth2 ? 0 : 1);
054        }
055    }
056
057    private static class GlobalCardRatioComparator implements Comparator<AstClafer> {
058
059        private final Map<AstClafer, Card> globalCards;
060
061        GlobalCardRatioComparator(Map<AstClafer, Card> globalCards) {
062            this.globalCards = Check.notNull(globalCards);
063        }
064
065        @Override
066        public int compare(AstClafer o1, AstClafer o2) {
067            Card card1 = notNull(o1 + " Global card not analyzed yet", globalCards.get(o1));
068            Card card2 = notNull(o2 + " Global card not analyzed yet", globalCards.get(o2));
069            double ratio1 = ((double) card1.getLow()) / ((double) card1.getHigh());
070            double ratio2 = ((double) card2.getLow()) / ((double) card2.getHigh());
071            return (ratio1 > ratio2) ? -1 : ((ratio1 == ratio2) ? 0 : 1);
072        }
073    }
074}
