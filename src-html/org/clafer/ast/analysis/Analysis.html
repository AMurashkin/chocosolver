073    private static Map<AstConcreteClafer, Card> buildCardMap(List<AstClafer> clafers) {
074        Map<AstConcreteClafer, Card> cardMap = new HashMap<AstConcreteClafer, Card>();
075        for (AstClafer clafer : clafers) {
076            if (clafer instanceof AstConcreteClafer) {
077                AstConcreteClafer concreteClafer = (AstConcreteClafer) clafer;
078                cardMap.put(concreteClafer, concreteClafer.getCard());
079            }
080        }
081        return cardMap;
082    }
083
084    private static List<AstClafer> append(List<? extends AstClafer> abstractClafers, List<? extends AstConcreteClafer> concreteClafers) {
085        List<AstClafer> clafers = new ArrayList<AstClafer>(abstractClafers.size() + concreteClafers.size());
086        clafers.addAll(abstractClafers);
087        clafers.addAll(concreteClafers);
088        return clafers;
089    }

091    public static Analysis analyze(AstModel model, Scope scope, Analyzer... analyzers) {
092        Analysis analysis = new Analysis(model, scope);
093        for (Analyzer analyzer : analyzers) {
094            analysis = analyzer.analyze(analysis);
095        }
096        return analysis;
097    }

099    private <T> T notNull(String analysisName, T t) {
100        if (t == null) {
101            throw new AnalysisException(analysisName + " not yet analyzed.");
102        }
103        return t;
104    }
105
106    private <T> T notNull(String key, String analysisName, T t) {
107        if (t == null) {
108            throw new AnalysisException(analysisName + " for " + key + " not yet analyzed.");
109        }
110        return t;
111    }
112
113    private <T> T notNull(AstClafer key, String analysisName, T t) {
114        return notNull(key.getName(), analysisName, t);
115    }

215    public Pair<AstConcreteClafer, Integer> getConcreteId(AstClafer clafer, int id) {
216        AstClafer sub = clafer;
217        int curId = id;
218        while (sub instanceof AstAbstractClafer) {
219            Offsets offset = getOffsets((AstAbstractClafer) sub);
220            sub = offset.getClafer(curId);
221            curId -= offset.getOffset(sub);
222        }
223        return new Pair<AstConcreteClafer, Integer>((AstConcreteClafer) sub, curId);
224    }
225
226    public Pair<AstClafer, Integer> getSubId(AstAbstractClafer clafer, int id) {
227        Offsets offsets = getOffsets(clafer);
228        AstClafer sub = offsets.getClafer(id);
229        return new Pair<AstClafer, Integer>(sub, id - offsets.getOffset(sub));
230    }

232    public Pair<AstAbstractClafer, Integer> getSuperId(AstClafer clafer, int id) {
233        assert clafer.hasSuperClafer();
234        int offset = getOffsets(clafer.getSuperClafer()).getOffset(clafer);
235        return new Pair<AstAbstractClafer, Integer>(clafer.getSuperClafer(), id + offset);
236    }
237
238    public List<Pair<AstAbstractClafer, Integer>> getSuperIds(AstClafer clafer, int id) {
239        List<Pair<AstAbstractClafer, Integer>> superIds = new ArrayList<Pair<AstAbstractClafer, Integer>>();
240        AstClafer sup = clafer;
241        int curId = id;
242        while (sup.hasSuperClafer()) {
243            Pair<AstAbstractClafer, Integer> superId = getSuperId(sup, curId);
244            superIds.add(superId);
245            sup = superId.getFst();
246            curId = superId.getSnd().intValue();
247        }
248        return superIds;
249    }

255    public List<Pair<AstClafer, Integer>> getHierarcyIds(AstClafer clafer, int id) {
256        List<Pair<AstClafer, Integer>> superIds = new ArrayList<Pair<AstClafer, Integer>>();
257        superIds.add(new Pair<AstClafer, Integer>(clafer, id));
258        AstClafer sup = clafer;
259        int curId = id;
260        while (sup.hasSuperClafer()) {
261            Pair<AstAbstractClafer, Integer> superId = getSuperId(sup, curId);
262            superIds.add(new Pair<AstClafer, Integer>(superId));
263            sup = superId.getFst();
264            curId = superId.getSnd().intValue();
265        }
266        return superIds;
267    }

269    public Pair<AstRef, Integer> getInheritedRefId(AstClafer clafer, int id) {
270        AstClafer sup = clafer;
271        int curId = id;
272        do {
273            if (sup.hasRef()) {
274                return new Pair<AstRef, Integer>(sup.getRef(), curId);
275            }
276            if (sup.hasSuperClafer()) {
277                curId += getOffsets(sup.getSuperClafer()).getOffset(sup);
278            }
279            sup = sup.getSuperClafer();
280        } while (sup != null);
281        return null;
282    }
