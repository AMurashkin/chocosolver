001package org.clafer.ast.compiler;
002
003import java.util.Map;
004import org.clafer.common.Check;
005import org.clafer.ast.analysis.Analysis;
006import org.clafer.ast.AstClafer;
007import org.clafer.ast.AstConstraint;
008import org.clafer.ast.AstException;
009import org.clafer.ast.AstModel;
010import org.clafer.ast.AstRef;
011import org.clafer.collection.Pair;
012import org.clafer.ir.IrBoolVar;
013import org.clafer.ir.IrIntVar;
014import org.clafer.ir.IrSetVar;
015
016/**
017 *
018 * @author jimmy
019 */
020public class AstSolutionMap {
021
022    private final AstModel model;
023    private final Map<AstClafer, IrSetVar[]> siblingVars;
024    private final Map<AstRef, IrIntVar[]> refVars;
025    private final Pair<AstConstraint, IrBoolVar>[] softVars;
026    private final Analysis analysis;
027
028    AstSolutionMap(AstModel model,
029            Map<AstClafer, IrSetVar[]> sibling,
030            Map<AstRef, IrIntVar[]> refVars,
031            Pair<AstConstraint, IrBoolVar>[] softVars,
032            Analysis analysis) {
033        this.model = Check.notNull(model);
034        this.siblingVars = Check.notNull(sibling);
035        this.refVars = Check.notNull(refVars);
036        this.softVars = Check.noNulls(softVars);
037        this.analysis = analysis;
038    }
039
040    public AstModel getModel() {
041        return model;
042    }
043
044    public Analysis getAnalysis() {
045        return analysis;
046    }
047
048    public IrSetVar[] getSiblingVars(AstClafer clafer) {
049        return notNull(clafer + " not part of the AST solution", siblingVars.get(clafer));
050    }
051
052    public IrIntVar[] getRefVars(AstRef ref) {
053        return notNull(ref + " not part of the AST solution", refVars.get(ref));
054    }
055
056    /**
057     * Returns the soft variables and their corresponding constraints.
058     *
059     * @return the soft variables and their corresponding constraints
060     */
061    public Pair<AstConstraint, IrBoolVar>[] getSoftVars() {
062        return softVars;
063    }
064
065    private static <T> T notNull(String message, T t) {
066        if (t == null) {
067            throw new AstException(message);
068        }
069        return t;
070    }
071}
