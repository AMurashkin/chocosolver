001package org.clafer.ast.compiler;
002
003import org.clafer.common.Check;
004import org.clafer.ast.analysis.Analysis;
005import org.clafer.ast.AstAbstractClafer;
006import org.clafer.ast.AstClafer;
007import org.clafer.ast.AstConstraint;
008import org.clafer.ast.AstException;
009import org.clafer.ast.AstModel;
010import org.clafer.ast.AstRef;
011import org.clafer.collection.Pair;
012import org.clafer.collection.ReadMap;
013import org.clafer.ir.IrBoolVar;
014import org.clafer.ir.IrIntVar;
015import org.clafer.ir.IrSetVar;
016
017/**
018 *
019 * @author jimmy
020 */
021public class AstSolutionMap {
022
023    private final AstModel model;
024    private final ReadMap<AstClafer, IrSetVar[]> childrenVars;
025    private final ReadMap<AstRef, IrIntVar[]> refVars;
026    private final Pair<AstConstraint, IrBoolVar>[] softVars;
027    private final Analysis analysis;
028
029    AstSolutionMap(AstModel model,
030            ReadMap<AstClafer, IrSetVar[]> childrenVars,
031            ReadMap<AstRef, IrIntVar[]> refVars,
032            Pair<AstConstraint, IrBoolVar>[] softVars,
033            Analysis analysis) {
034        this.model = Check.notNull(model);
035        this.childrenVars = childrenVars.readOnly();
036        this.refVars = refVars.readOnly();
037        this.softVars = Check.noNulls(softVars);
038        this.analysis = analysis;
039    }
040
041    public AstModel getModel() {
042        return model;
043    }
044
045    public Analysis getAnalysis() {
046        return analysis;
047    }
048
049    public IrSetVar[] getChildrenVars(AstClafer clafer) {
050        return notNull(clafer + " not part of the AST solution", childrenVars.get(clafer));
051    }
052
053    public IrIntVar[] getRefVars(AstRef ref) {
054        return notNull(ref + " not part of the AST solution", refVars.get(ref));
055    }
056
057    /**
058     * Returns the soft variables and their corresponding constraints.
059     *
060     * @return the soft variables and their corresponding constraints
061     */
062    public Pair<AstConstraint, IrBoolVar>[] getSoftVars() {
063        return softVars;
064    }
065
066    private static <T> T notNull(String message, T t) {
067        if (t == null) {
068            throw new AstException(message);
069        }
070        return t;
071    }
072}
