131    public static IrDomain enumDomain(TIntHashSet values) {
132        switch (values.size()) {
133            case 0:
134                return EmptyDomain;
135            case 1:
136                int value = values.iterator().next();
137                return boundDomain(value, value);
138            default:
139                int[] array = values.toArray();
140                Arrays.sort(array);
141                // If the values are over a contiguous interval, then return a bound domain.
142                int low = array[0];
143                for (int i = 1; i < array.length; i++) {
144                    if (low + i != array[i]) {
145                        return new IrEnumDomain(array);
146                    }
147                }
148                // A contigious interval.
149                return new IrBoundDomain(low, array[array.length - 1]);
150        }

183    public static IrBoolExpr and(IrBoolExpr... operands) {
184        List<IrBoolExpr> flatten = new ArrayList<IrBoolExpr>(operands.length);
185        for (IrBoolExpr operand : operands) {
186            flattenAnd(operand, flatten);
187        }
188        List<IrBoolExpr> filter = new ArrayList<IrBoolExpr>(flatten.size());
189        for (IrBoolExpr operand : flatten) {
190            if (IrUtil.isFalse(operand)) {
191                return $(False);
192            }
193            if (!IrUtil.isTrue(operand)) {
194                filter.add(operand);
195            }
196        }
197        switch (filter.size()) {
198            case 0:
199                return $(True);
200            case 1:
201                return filter.get(0);
202            default:
203                return new IrAnd(filter.toArray(new IrBoolExpr[filter.size()]), BoolDomain);
204        }
205    }

207    private static void flattenAnd(IrBoolExpr expr, List<IrBoolExpr> flatten) {
208        if (expr instanceof IrAnd) {
209            IrAnd and = (IrAnd) expr;
210            for (IrBoolExpr subexpr : and.getOperands()) {
211                flattenAnd(subexpr, flatten);
212            }
213        } else {
214            flatten.add(expr);
215        }
216    }

222    public static IrBoolExpr lone(IrBoolExpr... operands) {
223        List<IrBoolExpr> filter = new ArrayList<IrBoolExpr>(operands.length);
224        int count = 0;
225        for (IrBoolExpr operand : operands) {

228                if (count > 1) {
229                    return $(False);
230                }
231            }
232            if (!IrUtil.isFalse(operand)) {
233                filter.add(operand);
234            }
235        }
236        switch (filter.size()) {
237            case 0:
238                return $(True);
239            case 1:
240                return filter.get(0);
241            default:
242                return new IrLone(filter.toArray(new IrBoolExpr[filter.size()]), BoolDomain);
243        }
244    }

250    public static IrBoolExpr one(IrBoolExpr... operands) {
251        List<IrBoolExpr> filter = new ArrayList<IrBoolExpr>(operands.length);
252        int count = 0;
253        for (IrBoolExpr operand : operands) {
254            if (IrUtil.isTrue(operand)) {
255                count++;
256                if (count > 1) {
257                    return $(False);
258                }
259            }
260            if (!IrUtil.isFalse(operand)) {
261                filter.add(operand);
262            }
263        }
264        switch (filter.size()) {
265            case 0:
266                return $(count == 1 ? True : False);
267            case 1:
268                return filter.get(0);
269            case 2:
270                return xor(filter.get(0), filter.get(1));
271            default:
272                return new IrOne(filter.toArray(new IrBoolExpr[filter.size()]), BoolDomain);
273        }
274    }

315    public static IrBoolExpr implies(IrBoolExpr antecedent, IrBoolExpr consequent) {
316        if (IrUtil.isTrue(antecedent)) {
317            return consequent;
318        }
319        if (IrUtil.isFalse(antecedent)) {
320            return $(True);
321        }
322        if (IrUtil.isTrue(consequent)) {
323            return $(True);
324        }
325        if (IrUtil.isFalse(consequent)) {
326            return not(antecedent);
327        }
328        return new IrImplies(antecedent, consequent, BoolDomain);
329    }

347    public static IrBoolExpr ifThenElse(IrBoolExpr antecedent, IrBoolExpr consequent, IrBoolExpr alternative) {
348        if (IrUtil.isTrue(antecedent)) {
349            return consequent;
350        }
351        if (IrUtil.isFalse(antecedent)) {
352            return alternative;
353        }
354        if (IrUtil.isTrue(consequent)) {
355            return or(antecedent, alternative);
356        }
357        if (IrUtil.isFalse(consequent)) {
358            return and(antecedent.negate(), alternative);
359        }
360        if (IrUtil.isTrue(alternative)) {
361            return or(antecedent.negate(), consequent);
362        }
363        if (IrUtil.isFalse(alternative)) {
364            return and(antecedent, consequent);
365        }
366        return new IrIfThenElse(antecedent, consequent, alternative, BoolDomain);
367    }

370        if (left.equals(right)) {
371            return $(True);
372        }
373        if (IrUtil.isTrue(left)) {
374            return right;
375        }
376        if (IrUtil.isFalse(left)) {
377            return not(right);
378        }
379        if (IrUtil.isTrue(right)) {
380            return left;
381        }
382        if (IrUtil.isFalse(right)) {
383            return not(left);
384        }
385        return new IrIfOnlyIf(left, right, BoolDomain);

404    public static IrBoolExpr within(IrIntExpr var, IrDomain range) {
405        IrDomain domain = var.getDomain();
406        if (range.isBounded()
407                && domain.getLowBound() >= range.getLowBound()
408                && domain.getHighBound() <= range.getHighBound()) {
409            return $(True);
410        }
411        if (domain.getLowBound() > range.getHighBound()
412                || domain.getHighBound() < range.getLowBound()) {
413            return $(False);
414        }
415        return new IrWithin(var, range, BoolDomain);
416    }

444    public static IrBoolExpr compare(IrIntExpr left, IrCompare.Op op, IrIntExpr right) {
445        IrDomain leftDomain = left.getDomain();
446        IrDomain rightDomain = right.getDomain();
447        switch (op) {
448            case Equal:
449                if (left.equals(right)) {
450                    return $(True);
451                }
452                if (leftDomain.size() == 1 && rightDomain.size() == 1) {
453                    return $(constant(leftDomain.getLowBound() == rightDomain.getLowBound()));
454                }
455                if (isBoolDomain(left.getDomain()) && isBoolDomain(right.getDomain())) {
456                    return ifOnlyIf(asBool(left), asBool(right));
457                }
458                break;
459            case NotEqual:
460                if (left.equals(right)) {
461                    return $(False);
462                }
463                if (leftDomain.size() == 1 && rightDomain.size() == 1) {
464                    return $(constant(leftDomain.getLowBound() != rightDomain.getLowBound()));
465                }
466                if (isBoolDomain(left.getDomain()) && isBoolDomain(right.getDomain())) {
467                    return xor(asBool(left), asBool(right));
468                }
469                break;
470            case LessThan:
471                if (left.equals(right)) {
472                    return $(False);
473                }
474                if (leftDomain.getHighBound() < rightDomain.getLowBound()) {
475                    return $(True);
476                }
477                if (leftDomain.getLowBound() >= rightDomain.getHighBound()) {
478                    return $(False);
479                }
480                if (isBoolDomain(left.getDomain()) && isBoolDomain(right.getDomain())) {
481                    return not(implies(asBool(right), asBool(left)));
482                }
483                break;
484            case LessThanEqual:
485                if (left.equals(right)) {
486                    return $(True);
487                }
488                if (leftDomain.getHighBound() <= rightDomain.getLowBound()) {
489                    return $(True);
490                }
491                if (leftDomain.getLowBound() > rightDomain.getHighBound()) {
492                    return $(False);
493                }
494                if (isBoolDomain(left.getDomain()) && isBoolDomain(right.getDomain())) {
495                    return implies(asBool(left), asBool(right));
496                }
497                break;
498            case GreaterThan:
499                if (left.equals(right)) {
500                    return $(False);
501                }
502                if (leftDomain.getLowBound() > rightDomain.getHighBound()) {
503                    return $(True);
504                }
505                if (leftDomain.getHighBound() <= rightDomain.getLowBound()) {
506                    return $(False);
507                }
508                if (isBoolDomain(left.getDomain()) && isBoolDomain(right.getDomain())) {
509                    return not(implies(asBool(left), asBool(right)));
510                }
511                break;
512            case GreaterThanEqual:
513                if (left.equals(right)) {
514                    return $(True);
515                }
516                if (leftDomain.getLowBound() >= rightDomain.getHighBound()) {
517                    return $(True);
518                }
519                if (leftDomain.getHighBound() < rightDomain.getLowBound()) {
520                    return $(False);
521                }
522                if (isBoolDomain(left.getDomain()) && isBoolDomain(right.getDomain())) {
523                    return implies(asBool(right), asBool(left));
524                }
525                break;
526            default:
527                throw new IllegalArgumentException("Unknown op: " + op);
528        }
529        return new IrCompare(left, op, right, BoolDomain);
530    }

544    public static IrBoolExpr equality(IrSetExpr left, IrSetTest.Op op, IrSetExpr right) {
545        switch (op) {
546            case Equal:
547                int[] constant = IrUtil.getConstant(left);
548                if (constant != null) {
549                    if (constant.length == 0) {
550                        return equal(card(right), 0);
551                    }
552                    if (constant.length == right.getEnv().size()) {
553                        if (IrUtil.containsAll(constant, right.getEnv())) {
554                            return equal(card(right), constant.length);
555                        }
556                    }
557                }
558                constant = IrUtil.getConstant(right);
559                if (constant != null) {
560                    if (constant.length == 0) {
561                        return equal(card(left), 0);
562                    }
563                    if (constant.length == left.getEnv().size()) {
564                        if (IrUtil.containsAll(constant, left.getEnv())) {
565                            return equal(card(left), constant.length);
566                        }
567                    }
568                }
569                break;
570            case NotEqual:
571                constant = IrUtil.getConstant(left);
572                if (constant != null) {
573                    if (constant.length == 0) {
574                        return equal(card(right), 0);
575                    }
576                    if (constant.length == right.getEnv().size()) {
577                        if (IrUtil.containsAll(constant, right.getEnv())) {
578                            return notEqual(card(right), constant.length);
579                        }
580                    }
581                }
582                constant = IrUtil.getConstant(right);
583                if (constant != null) {
584                    if (constant.length == 0) {
585                        return equal(card(left), 0);
586                    }
587                    if (constant.length == left.getEnv().size()) {
588                        if (IrUtil.containsAll(constant, left.getEnv())) {
589                            return notEqual(card(left), constant.length);
590                        }
591                    }
592                }
593                break;
594            default:
595                throw new IllegalArgumentException();
596        }
597        return new IrSetTest(left, op, right, BoolDomain);
598    }

668    public static IrBoolExpr member(IrIntExpr element, IrSetExpr set) {
669        if (IrUtil.isSubsetOf(element.getDomain(), set.getKer())) {
670            return $(True);
671        }
672        if (!IrUtil.intersects(element.getDomain(), set.getEnv())) {
673            return $(False);
674        }
675        if (IrUtil.isConstant(set)) {
676            return within(element, set.getEnv());
677        }
678        return new IrMember(element, set, BoolDomain);
679    }

709        Integer constant = IrUtil.getConstant(expr);
710        if (constant != null) {

719            IrIntCast intCast = (IrIntCast) expr;
720            return intCast.getExpr();
721        }
722        return new IrBoolCast(flipped, expr, BoolDomain);

1009        for (int i = 1; i < subtrahends.length; i++) {

1155        int sum = Util.sum(set.getKer().iterator());
1156        int count = set.getKer().size();
1157
1158        // Calculate low
1159        int low = sum;
1160        int lowCount = count;
1161        TIntIterator envIter = set.getEnv().iterator();
1162        while (lowCount < set.getCard().getHighBound() && envIter.hasNext()) {
1163            int env = envIter.next();
1164            if (env >= 0 && lowCount >= set.getCard().getLowBound()) {
1165                break;
1166            }
1167            if (!set.getKer().contains(env)) {
1168                low += env;
1169                lowCount++;
1170            }
1171        }
1172
1173        // Calculate high
1174        int high = sum;
1175        int highCount = count;
1176        envIter = set.getEnv().iterator(false);
1177        while (highCount < set.getCard().getHighBound() && envIter.hasNext()) {
1178            int env = envIter.next();
1179            if (env <= 0 && highCount >= set.getCard().getLowBound()) {
1180                break;
1181            }
1182            if (!set.getKer().contains(env)) {
1183                high += env;
1184                highCount++;
1185            }
1186        }
1187
1188        return new IrSetSum(set, boundDomain(low, high));

1297    /**
1298     * Relational join.
1299     *
1300     * Union{for all i in take} children[i]
1301     *
1302     * @param take
1303     * @param children
1304     * @return the join expression take.children
1305     */
1306    public static IrSetExpr joinRelation(IrSetExpr take, IrSetExpr[] children, boolean injective) {
1307        IrSetExpr[] $children = children;
1308        if (take.getEnv().getHighBound() + 1 < $children.length) {
1309            $children = Arrays.copyOf(children, take.getEnv().getHighBound() + 1);
1310        }
1311
1312        int[] constant = IrUtil.getConstant(take);
1313        if (constant != null) {
1314            IrSetExpr[] to = new IrSetExpr[constant.length];
1315            for (int i = 0; i < to.length; i++) {
1316                to[i] = $children[constant[i]];
1317            }
1318            return union(to);
1319        }
1320
1321        // Compute env
1322        TIntIterator iter = take.getEnv().iterator();
1323        IrDomain env;
1324        if (iter.hasNext()) {
1325            IrDomain domain = $children[iter.next()].getEnv();
1326            while (iter.hasNext()) {
1327                domain = IrUtil.union(domain, $children[iter.next()].getEnv());
1328            }
1329            env = domain;
1330        } else {
1331            env = Irs.EmptyDomain;
1332        }
1333
1334        // Compute ker
1335        iter = take.getKer().iterator();
1336        IrDomain ker;
1337        if (iter.hasNext()) {
1338            IrDomain domain = $children[iter.next()].getKer();
1339            while (iter.hasNext()) {
1340                domain = IrUtil.union(domain, $children[iter.next()].getKer());
1341            }
1342            ker = domain;
1343        } else {
1344            ker = Irs.EmptyDomain;
1345        }
1346
1347        // Compute card
1348        IrDomain takeEnv = take.getEnv();
1349        IrDomain takeKer = take.getKer();
1350        IrDomain takeCard = take.getCard();
1351        int index = 0;
1352        int[] childrenLowCards = new int[takeEnv.size() - takeKer.size()];
1353        int[] childrenHighCards = new int[takeEnv.size() - takeKer.size()];
1354        int cardLow = 0, cardHigh = 0;
1355
1356        iter = takeEnv.iterator();
1357        while (iter.hasNext()) {
1358            int val = iter.next();
1359            IrDomain childDomain = $children[val].getCard();
1360            if (takeKer.contains(val)) {
1361                cardLow += childDomain.getLowBound();
1362                cardHigh += childDomain.getHighBound();
1363            } else {
1364                childrenLowCards[index] = childDomain.getLowBound();
1365                childrenHighCards[index] = childDomain.getHighBound();
1366                index++;
1367            }
1368        }
1369        assert index == childrenLowCards.length;
1370        assert index == childrenHighCards.length;
1371
1372        Arrays.sort(childrenLowCards);
1373        Arrays.sort(childrenHighCards);
1374
1375        for (int i = 0; i < takeCard.getLowBound() - takeKer.size(); i++) {
1376            cardLow += childrenLowCards[i];
1377        }
1378        for (int i = 0; i < takeCard.getHighBound() - takeKer.size(); i++) {
1379            cardHigh += childrenHighCards[childrenHighCards.length - 1 - i];
1380        }
1381        cardLow = Math.max(cardLow, ker.size());
1382        cardHigh = Math.min(cardHigh, env.size());
1383        IrDomain card = boundDomain(cardLow, cardHigh);
1384
1385        return new IrJoinRelation(take, $children, env, ker, card, injective);
1386    }

1389    public static IrSetExpr joinFunction(IrSetExpr take, IrIntExpr[] refs, Integer globalCardinality) {
1390        int[] constant = IrUtil.getConstant(take);
1391        if (constant != null) {
1392            IrIntExpr[] to = new IrIntExpr[constant.length];
1393            for (int i = 0; i < to.length; i++) {
1394                to[i] = refs[constant[i]];
1395            }
1396            return arrayToSet(to, globalCardinality);
1397        }
1398
1399        // Compute env
1400        TIntIterator iter = take.getEnv().iterator();
1401        IrDomain env;
1402        if (iter.hasNext()) {
1403            IrDomain domain = refs[iter.next()].getDomain();
1404            while (iter.hasNext()) {
1405                domain = IrUtil.union(domain, refs[iter.next()].getDomain());
1406            }
1407            env = domain;
1408        } else {
1409            env = Irs.EmptyDomain;
1410        }
1411
1412        // Compute ker
1413        iter = take.getKer().iterator();
1414        TIntHashSet values = new TIntHashSet(0);
1415        while (iter.hasNext()) {
1416            Integer constantRef = IrUtil.getConstant(refs[iter.next()]);
1417            if (constantRef != null) {
1418                values.add(constantRef.intValue());
1419            }
1420        }
1421        IrDomain ker = values.isEmpty() ? Irs.EmptyDomain : new IrEnumDomain(values.toArray());
1422
1423        // Compute card
1424        IrDomain takeCard = take.getCard();
1425        int lowTakeCard = takeCard.getLowBound();
1426        int highTakeCard = takeCard.getHighBound();
1427        IrDomain card = lowTakeCard == 0
1428                ? boundDomain(Math.max(0, ker.size()), Math.min(highTakeCard, env.size()))
1429                : boundDomain(Math.max(1, ker.size()), Math.min(highTakeCard, env.size()));
1430
1431        return new IrJoinFunction(take, refs, env, ker, card, globalCardinality);
1432    }

1434    public static IrSetExpr difference(IrSetExpr minuend, IrSetExpr subtrahend) {
1435        IrDomain env = IrUtil.difference(minuend.getEnv(), subtrahend.getKer());
1436        IrDomain ker = IrUtil.difference(minuend.getKer(), subtrahend.getEnv());
1437        int low = Math.max(0, minuend.getCard().getLowBound() - subtrahend.getCard().getHighBound());
1438        int high = minuend.getCard().getHighBound();
1439        IrDomain card = boundDomain(Math.max(low, ker.size()), Math.min(high, env.size()));
1440        return new IrSetDifference(minuend, subtrahend, env, ker, card);
1441    }

1443    public static IrSetExpr intersection(IrSetExpr... operands) {
1444        switch (operands.length) {
1445            case 0:
1446                return $(EmptySet);
1447            case 1:
1448                return operands[0];
1449            default:
1450                IrDomain env = IrUtil.intersectionEnvs(operands);
1451                IrDomain ker = IrUtil.intersectionKers(operands);
1452                int low = 0;
1453                int high = operands[0].getCard().getHighBound();
1454                for (int i = 1; i < operands.length; i++) {
1455                    high = Math.max(high, operands[0].getCard().getHighBound());
1456                }
1457                IrDomain card = boundDomain(
1458                        Math.max(low, ker.size()),
1459                        Math.min(high, env.size()));
1460                return new IrSetIntersection(operands, env, ker, card);
1461        }
1462    }

1464    public static IrSetExpr union(IrSetExpr... operands) {
1465        List<IrSetExpr> flatten = new ArrayList<IrSetExpr>(operands.length);
1466        for (IrSetExpr operand : operands) {
1467            flattenUnion(operand, flatten);
1468        }
1469        operands = flatten.toArray(new IrSetExpr[flatten.size()]);
1470        switch (operands.length) {
1471            case 0:
1472                return $(EmptySet);
1473            case 1:
1474                return operands[0];
1475            default:
1476                IrDomain env = IrUtil.unionEnvs(operands);
1477                IrDomain ker = IrUtil.unionKers(operands);
1478                IrDomain operandCard = operands[0].getCard();
1479                int low = operandCard.getLowBound();
1480                int high = operandCard.getHighBound();
1481                for (int i = 1; i < operands.length; i++) {
1482                    operandCard = operands[i].getCard();
1483                    low = Math.max(low, operandCard.getLowBound());
1484                    high += operandCard.getHighBound();
1485                }
1486                IrDomain card = boundDomain(
1487                        Math.max(low, ker.size()),
1488                        Math.min(high, env.size()));
1489                return IrUtil.asConstant(new IrSetUnion(operands, env, ker, card));
1490        }
1491    }

1493    private static void flattenUnion(IrSetExpr expr, List<IrSetExpr> flatten) {
1494        if (expr instanceof IrSetUnion) {
1495            IrSetUnion union = (IrSetUnion) expr;
1496            for (IrSetExpr subexpr : union.getOperands()) {
1497                flattenUnion(subexpr, flatten);
1498            }
1499        } else {
1500            flatten.add(expr);
1501        }
1502    }

1504    public static IrSetExpr offset(IrSetExpr set, int offset) {
1505        if (offset == 0) {
1506            return set;
1507        }
1508        int[] constant = IrUtil.getConstant(set);
1509        if (constant != null) {
1510            for (int i = 0; i < constant.length; i++) {
1511                constant[i] += offset;
1512            }
1513            return $(constant(constant));
1514        }
1515        IrDomain env = IrUtil.offset(set.getEnv(), offset);
1516        IrDomain ker = IrUtil.offset(set.getKer(), offset);
1517        IrDomain card = set.getCard();
1518        return new IrOffset(set, offset, env, ker, card);
1519    }

1521    public static IrSetExpr ternary(IrBoolExpr antecedent, IrSetExpr consequent, IrSetExpr alternative) {
1522        if (IrUtil.isTrue(antecedent)) {
1523            return consequent;
1524        }
1525        if (IrUtil.isFalse(antecedent)) {
1526            return alternative;
1527        }
1528        if (consequent.equals(alternative)) {
1529            return consequent;
1530        }
1531        int[] consequentConstant = IrUtil.getConstant(consequent);
1532        int[] alternativeConstant = IrUtil.getConstant(alternative);
1533        if (consequentConstant != null && alternativeConstant != null && Arrays.equals(consequentConstant, alternativeConstant)) {
1534            return $(constant(consequentConstant));
1535        }
1536        IrDomain env = IrUtil.union(consequent.getEnv(), alternative.getEnv());
1537        IrDomain ker = IrUtil.intersection(consequent.getKer(), alternative.getKer());
1538        IrDomain card = IrUtil.union(consequent.getCard(), alternative.getCard());
1539        return new IrSetTernary(antecedent, consequent, alternative, env, ker, card);
1540    }
