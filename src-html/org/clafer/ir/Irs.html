127    public static IrDomain enumDomain(TIntHashSet values) {
128        switch (values.size()) {
129            case 0:
130                return EmptyDomain;
131            case 1:
132                int value = values.iterator().next();
133                return boundDomain(value, value);
134            default:
135                int[] array = values.toArray();
136                Arrays.sort(array);
137                // If the values are over a contiguous interval, then return a bound domain.
138                int low = array[0];
139                for (int i = 1; i < array.length; i++) {
140                    if (low + i != array[i]) {
141                        return new IrEnumDomain(array);
142                    }
143                }
144                // A contigious interval.
145                return new IrBoundDomain(low, array[array.length - 1]);
146        }

179    public static IrBoolExpr and(IrBoolExpr... operands) {
180        List<IrBoolExpr> flatten = new ArrayList<IrBoolExpr>(operands.length);
181        for (IrBoolExpr operand : operands) {
182            flattenAnd(operand, flatten);
183        }
184        List<IrBoolExpr> filter = new ArrayList<IrBoolExpr>(flatten.size());
185        for (IrBoolExpr operand : flatten) {
186            if (IrUtil.isFalse(operand)) {
187                return $(False);
188            }
189            if (!IrUtil.isTrue(operand)) {
190                filter.add(operand);
191            }
192        }
193        switch (filter.size()) {
194            case 0:
195                return $(True);
196            case 1:
197                return filter.get(0);
198            default:
199                return new IrAnd(filter.toArray(new IrBoolExpr[filter.size()]), BoolDomain);
200        }
201    }

203    private static void flattenAnd(IrBoolExpr expr, List<IrBoolExpr> flatten) {
204        if (expr instanceof IrAnd) {
205            IrAnd and = (IrAnd) expr;
206            for (IrBoolExpr subexpr : and.getOperands()) {
207                flattenAnd(subexpr, flatten);
208            }
209        } else {
210            flatten.add(expr);
211        }
212    }

228            if (!IrUtil.isFalse(operand)) {
229                filter.add(operand);
230            }
231        }
232        switch (filter.size()) {
233            case 0:
234                return $(True);
235            case 1:
236                return filter.get(0);
237            default:
238                return new IrLone(filter.toArray(new IrBoolExpr[filter.size()]), BoolDomain);

246    public static IrBoolExpr one(IrBoolExpr... operands) {
247        List<IrBoolExpr> filter = new ArrayList<IrBoolExpr>(operands.length);
248        int count = 0;
249        for (IrBoolExpr operand : operands) {
250            if (IrUtil.isTrue(operand)) {
251                count++;
252                if (count > 1) {
253                    return $(False);
254                }
255            }
256            if (!IrUtil.isFalse(operand)) {
257                filter.add(operand);
258            }
259        }
260        switch (filter.size()) {
261            case 0:
262                return $(count == 1 ? True : False);
263            case 1:
264                return filter.get(0);
265            case 2:
266                return xor(filter.get(0), filter.get(1));
267            default:
268                return new IrOne(filter.toArray(new IrBoolExpr[filter.size()]), BoolDomain);
269        }

311    public static IrBoolExpr implies(IrBoolExpr antecedent, IrBoolExpr consequent) {
312        if (IrUtil.isTrue(antecedent)) {
313            return consequent;
314        }
315        if (IrUtil.isFalse(antecedent)) {
316            return $(True);
317        }
318        if (IrUtil.isTrue(consequent)) {
319            return $(True);
320        }
321        if (IrUtil.isFalse(consequent)) {
322            return not(antecedent);
323        }
324        return new IrImplies(antecedent, consequent, BoolDomain);
325    }

343    public static IrBoolExpr ifThenElse(IrBoolExpr antecedent, IrBoolExpr consequent, IrBoolExpr alternative) {
344        if (IrUtil.isTrue(antecedent)) {
345            return consequent;
346        }
347        if (IrUtil.isFalse(antecedent)) {
348            return alternative;
349        }
350        if (IrUtil.isTrue(consequent)) {
351            return or(antecedent, alternative);
352        }
353        if (IrUtil.isFalse(consequent)) {
354            return and(antecedent.negate(), alternative);
355        }
356        if (IrUtil.isTrue(alternative)) {
357            return or(antecedent.negate(), consequent);
358        }
359        if (IrUtil.isFalse(alternative)) {
360            return and(antecedent, consequent);
361        }
362        return new IrIfThenElse(antecedent, consequent, alternative, BoolDomain);
363    }

397    public static IrBoolExpr within(IrIntExpr var, IrDomain range) {
398        IrDomain domain = var.getDomain();
399        if (range.isBounded()
400                && domain.getLowBound() >= range.getLowBound()
401                && domain.getHighBound() <= range.getHighBound()) {
402            return $(True);
403        }
404        if (domain.getLowBound() > range.getHighBound()
405                || domain.getHighBound() < range.getLowBound()) {
406            return $(False);
407        }
408        return new IrWithin(var, range, BoolDomain);
409    }

437    public static IrBoolExpr compare(IrIntExpr left, IrCompare.Op op, IrIntExpr right) {
438        IrDomain leftDomain = left.getDomain();
439        IrDomain rightDomain = right.getDomain();
440        switch (op) {
441            case Equal:
442                if (leftDomain.size() == 1 && rightDomain.size() == 1) {
443                    return $(constant(leftDomain.getLowBound() == rightDomain.getLowBound()));
444                }
445                if (isBoolDomain(left.getDomain()) && isBoolDomain(right.getDomain())) {
446                    return ifOnlyIf(asBool(left), asBool(right));
447                }
448                break;
449            case NotEqual:
450                if (leftDomain.size() == 1 && rightDomain.size() == 1) {
451                    return $(constant(leftDomain.getLowBound() != rightDomain.getLowBound()));
452                }
453                if (isBoolDomain(left.getDomain()) && isBoolDomain(right.getDomain())) {
454                    return xor(asBool(left), asBool(right));
455                }
456                break;
457            case LessThan:
458                if (leftDomain.getHighBound() < rightDomain.getLowBound()) {
459                    return $(True);
460                }
461                if (leftDomain.getLowBound() >= rightDomain.getHighBound()) {
462                    return $(False);
463                }
464                if (isBoolDomain(left.getDomain()) && isBoolDomain(right.getDomain())) {
465                    return not(implies(asBool(right), asBool(left)));
466                }
467                break;
468            case LessThanEqual:
469                if (leftDomain.getHighBound() <= rightDomain.getLowBound()) {
470                    return $(True);
471                }
472                if (leftDomain.getLowBound() > rightDomain.getHighBound()) {
473                    return $(False);
474                }
475                if (isBoolDomain(left.getDomain()) && isBoolDomain(right.getDomain())) {
476                    return implies(asBool(left), asBool(right));
477                }
478                break;
479            case GreaterThan:
480                if (leftDomain.getLowBound() > rightDomain.getHighBound()) {
481                    return $(True);
482                }
483                if (leftDomain.getHighBound() <= rightDomain.getLowBound()) {
484                    return $(False);
485                }
486                if (isBoolDomain(left.getDomain()) && isBoolDomain(right.getDomain())) {
487                    return not(implies(asBool(left), asBool(right)));
488                }
489                break;
490            case GreaterThanEqual:
491                if (leftDomain.getLowBound() >= rightDomain.getHighBound()) {
492                    return $(True);
493                }
494                if (leftDomain.getHighBound() < rightDomain.getLowBound()) {
495                    return $(False);
496                }
497                if (isBoolDomain(left.getDomain()) && isBoolDomain(right.getDomain())) {
498                    return implies(asBool(right), asBool(left));
499                }
500                break;
501            default:
502                throw new IllegalArgumentException("Unknown op: " + op);
503        }
504        return new IrCompare(left, op, right, BoolDomain);
505    }

519    public static IrBoolExpr equality(IrSetExpr left, IrSetTest.Op op, IrSetExpr right) {
520        switch (op) {
521            case Equal:
522                int[] constant = IrUtil.getConstant(left);
523                if (constant != null) {
524                    if (constant.length == 0) {
525                        return equal(card(right), 0);
526                    }
527                    if (constant.length == right.getEnv().size()) {
528                        if (IrUtil.containsAll(constant, right.getEnv())) {
529                            return equal(card(right), constant.length);
530                        }
531                    }
532                }
533                constant = IrUtil.getConstant(right);
534                if (constant != null) {
535                    if (constant.length == 0) {
536                        return equal(card(left), 0);
537                    }
538                    if (constant.length == left.getEnv().size()) {
539                        if (IrUtil.containsAll(constant, left.getEnv())) {
540                            return equal(card(left), constant.length);
541                        }
542                    }
543                }
544                break;
545            case NotEqual:
546                constant = IrUtil.getConstant(left);
547                if (constant != null) {
548                    if (constant.length == 0) {
549                        return equal(card(right), 0);
550                    }
551                    if (constant.length == right.getEnv().size()) {
552                        if (IrUtil.containsAll(constant, right.getEnv())) {
553                            return notEqual(card(right), constant.length);
554                        }
555                    }
556                }
557                constant = IrUtil.getConstant(right);
558                if (constant != null) {
559                    if (constant.length == 0) {
560                        return equal(card(left), 0);
561                    }
562                    if (constant.length == left.getEnv().size()) {
563                        if (IrUtil.containsAll(constant, left.getEnv())) {
564                            return notEqual(card(left), constant.length);
565                        }
566                    }
567                }
568                break;
569            default:
570                throw new IllegalArgumentException();
571        }
572        return new IrSetTest(left, op, right, BoolDomain);
573    }

643    public static IrBoolExpr member(IrIntExpr element, IrSetExpr set) {
644        if (IrUtil.isSubsetOf(element.getDomain(), set.getKer())) {
645            return $(True);
646        }
647        if (!IrUtil.intersects(element.getDomain(), set.getEnv())) {
648            return $(False);
649        }
650        if (IrUtil.isConstant(set)) {
651            return within(element, set.getEnv());
652        }
653        return new IrMember(element, set, BoolDomain);
654    }

712    public static IrBoolExpr boolChannel(IrBoolExpr[] bools, IrSetExpr set) {
713        IrDomain env = set.getEnv();
714        IrDomain ker = set.getKer();
715        boolean entailed = true;
716        for (int i = 0; i < bools.length; i++) {
717            Boolean constant = IrUtil.getConstant(bools[i]);
718            if (Boolean.TRUE.equals(constant)) {
719                if (!env.contains(i)) {
720                    return $(False);
721                }
722                if (!ker.contains(i)) {
723                    entailed = false;
724                }
725            } else if (Boolean.FALSE.equals(constant)) {

1085    public static IrIntExpr sum(IrSetExpr set) {
1086        int sum = Util.sum(set.getKer().iterator());
1087        int count = set.getKer().size();
1088
1089        // Calculate low
1090        int low = sum;
1091        int lowCount = count;
1092        TIntIterator envIter = set.getEnv().iterator();
1093        while (lowCount < set.getCard().getHighBound() && envIter.hasNext()) {
1094            int env = envIter.next();
1095            if (env >= 0 && lowCount >= set.getCard().getLowBound()) {
1096                break;
1097            }
1098            if (!set.getKer().contains(env)) {
1099                low += env;
1100                lowCount++;
1101            }
1102        }
1103
1104        // Calculate high
1105        int high = sum;
1106        int highCount = count;
1107        envIter = set.getEnv().iterator(false);
1108        while (highCount < set.getCard().getHighBound() && envIter.hasNext()) {
1109            int env = envIter.next();
1110            if (env <= 0 && highCount >= set.getCard().getLowBound()) {
1111                break;
1112            }
1113            if (!set.getKer().contains(env)) {
1114                high += env;
1115                highCount++;
1116            }
1117        }
1118
1119        return new IrSetSum(set, boundDomain(low, high));

1227    /**
1228     * Relational join.
1229     *
1230     * Union{for all i in take} children[i]
1231     *
1232     * @param take
1233     * @param children
1234     * @return the join expression take.children
1235     */
1236    public static IrSetExpr joinRelation(IrSetExpr take, IrSetExpr[] children) {
1237        IrSetExpr[] $children = children;
1238        if (take.getEnv().getHighBound() + 1 < $children.length) {
1239            $children = Arrays.copyOf(children, take.getEnv().getHighBound() + 1);
1240        }
1241
1242        int[] constant = IrUtil.getConstant(take);
1243        if (constant != null) {
1244            IrSetExpr[] to = new IrSetExpr[constant.length];
1245            for (int i = 0; i < to.length; i++) {
1246                to[i] = $children[constant[i]];
1247            }
1248            return union(to);
1249        }
1250
1251        // Compute env
1252        TIntIterator iter = take.getEnv().iterator();
1253        IrDomain env;
1254        if (iter.hasNext()) {
1255            IrDomain domain = $children[iter.next()].getEnv();
1256            while (iter.hasNext()) {
1257                domain = IrUtil.union(domain, $children[iter.next()].getEnv());
1258            }
1259            env = domain;
1260        } else {
1261            env = Irs.EmptyDomain;
1262        }
1263
1264        // Compute ker
1265        iter = take.getKer().iterator();
1266        IrDomain ker;
1267        if (iter.hasNext()) {
1268            IrDomain domain = $children[iter.next()].getKer();
1269            while (iter.hasNext()) {
1270                domain = IrUtil.union(domain, $children[iter.next()].getKer());
1271            }
1272            ker = domain;
1273        } else {
1274            ker = Irs.EmptyDomain;
1275        }
1276
1277        // Compute card
1278        IrDomain takeEnv = take.getEnv();
1279        IrDomain takeKer = take.getKer();
1280        IrDomain takeCard = take.getCard();
1281        int index = 0;
1282        int[] childrenLowCards = new int[takeEnv.size() - takeKer.size()];
1283        int[] childrenHighCards = new int[takeEnv.size() - takeKer.size()];
1284        int cardLow = 0, cardHigh = 0;
1285
1286        iter = takeEnv.iterator();
1287        while (iter.hasNext()) {
1288            int val = iter.next();
1289            IrDomain childDomain = $children[val].getCard();
1290            if (takeKer.contains(val)) {
1291                cardLow += childDomain.getLowBound();
1292                cardHigh += childDomain.getHighBound();
1293            } else {
1294                childrenLowCards[index] = childDomain.getLowBound();
1295                childrenHighCards[index] = childDomain.getHighBound();
1296                index++;
1297            }
1298        }
1299        assert index == childrenLowCards.length;
1300        assert index == childrenHighCards.length;
1301
1302        Arrays.sort(childrenLowCards);
1303        Arrays.sort(childrenHighCards);
1304
1305        for (int i = 0; i < takeCard.getLowBound() - takeKer.size(); i++) {
1306            cardLow += childrenLowCards[i];
1307        }
1308        for (int i = 0; i < takeCard.getHighBound() - takeKer.size(); i++) {
1309            cardHigh += childrenHighCards[childrenHighCards.length - 1 - i];
1310        }
1311        cardLow = Math.max(cardLow, ker.size());
1312        cardHigh = Math.min(cardHigh, env.size());
1313        IrDomain card = boundDomain(cardLow, cardHigh);
1314
1315        return new IrJoinRelation(take, $children, env, ker, card);
1316    }

1376            case 1:
1377                return operands[0];
1378            default:
1379                IrDomain env = IrUtil.intersectionEnvs(operands);
1380                IrDomain ker = IrUtil.intersectionKers(operands);
1381                int low = 0;
1382                int high = operands[0].getCard().getHighBound();
1383                for (int i = 1; i < operands.length; i++) {
1384                    high = Math.max(high, operands[0].getCard().getHighBound());
1385                }
1386                IrDomain card = boundDomain(
1387                        Math.max(low, ker.size()),
1388                        Math.min(high, env.size()));
1389                return new IrSetIntersection(operands, env, ker, card);
1390        }
1391    }

1393    public static IrSetExpr union(IrSetExpr... operands) {
1394        switch (operands.length) {
1395            case 0:
1396                return $(EmptySet);
1397            case 1:
1398                return operands[0];
1399            default:
1400                IrDomain env = IrUtil.unionEnvs(operands);
1401                IrDomain ker = IrUtil.unionKers(operands);
1402                IrDomain operandCard = operands[0].getCard();
1403                int low = operandCard.getLowBound();
1404                int high = operandCard.getHighBound();
1405                for (int i = 1; i < operands.length; i++) {
1406                    operandCard = operands[i].getCard();
1407                    low = Math.max(low, operandCard.getLowBound());
1408                    high += operandCard.getHighBound();
1409                }
1410                IrDomain card = boundDomain(
1411                        Math.max(low, ker.size()),
1412                        Math.min(high, env.size()));
1413                return new IrSetUnion(operands, env, ker, card);
1414        }
1415    }

1417    public static IrSetExpr offset(IrSetExpr set, int offset) {
1418        if (offset == 0) {
1419            return set;
1420        }
1421        int[] constant = IrUtil.getConstant(set);
1422        if (constant != null) {
1423            for (int i = 0; i < constant.length; i++) {
1424                constant[i] += offset;
1425            }
1426            return $(constant(constant));
1427        }
1428        IrDomain env = IrUtil.offset(set.getEnv(), offset);
1429        IrDomain ker = IrUtil.offset(set.getKer(), offset);
1430        IrDomain card = set.getCard();
1431        return new IrOffset(set, offset, env, ker, card);
1432    }

1434    public static IrSetExpr ternary(IrBoolExpr antecedent, IrSetExpr consequent, IrSetExpr alternative) {
1435        if (IrUtil.isTrue(antecedent)) {
1436            return consequent;
1437        }
1438        if (IrUtil.isFalse(antecedent)) {
1439            return alternative;
1440        }
1441        if (consequent.equals(alternative)) {
1442            return consequent;
1443        }
1444        int[] consequentConstant = IrUtil.getConstant(consequent);
1445        int[] alternativeConstant = IrUtil.getConstant(alternative);
1446        if (consequentConstant != null && alternativeConstant != null && Arrays.equals(consequentConstant, alternativeConstant)) {
1447            return $(constant(consequentConstant));
1448        }
1449        IrDomain env = IrUtil.union(consequent.getEnv(), alternative.getEnv());
1450        IrDomain ker = IrUtil.intersection(consequent.getKer(), alternative.getKer());
1451        IrDomain card = IrUtil.union(consequent.getCard(), alternative.getCard());
1452        return new IrSetTernary(antecedent, consequent, alternative, env, ker, card);
1453    }
