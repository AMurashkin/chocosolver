001package org.clafer.ir;
002
003/**
004 * Boolean domain.
005 *
006 * @author jimmy
007 */
008public enum IrBoolDomain {
009
010    TrueDomain(true, false),
011    FalseDomain(false, true),
012    BoolDomain(true, true);
013    private final boolean hasTrue, hasFalse;
014
015    private IrBoolDomain(boolean hasTrue, boolean hasFalse) {
016        // At least one has to be true.
017        this.hasTrue = hasTrue;
018        this.hasFalse = hasFalse;
019    }
020
021    /**
022     * @param value test this value
023     * @return {@code true} if and only if the domain contains the {@code value},
024     *         {@code false} otherwise
025     */
026    public boolean contains(boolean value) {
027        return value ? hasTrue : hasFalse;
028    }
029
030    /**
031     * @return the size of the domain
032     */
033    public int size() {
034        return hasTrue && hasFalse ? 2 : 1;
035    }
036
037    /**
038     * Reverse the domain. Maps {0}->{1}, {1}->{0}, and {0,1}->{0,1}.
039     *
040     * @return the inverted domain
041     */
042    public IrBoolDomain invert() {
043        switch (this) {
044            case TrueDomain:
045                return FalseDomain;
046            case FalseDomain:
047                return TrueDomain;
048            case BoolDomain:
049                return BoolDomain;
050            default:
051                throw new IllegalStateException();
052        }
053    }
054}
