034    public static Boolean getConstant(IrBool b) {
035        switch (b.getDomain()) {
036            case TrueDomain:
037                return Boolean.TRUE;
038            case FalseDomain:
039                return Boolean.FALSE;
040            case BoolDomain:
041                return null;
042            default:
043                throw new IllegalArgumentException();
044        }
045    }

075    public static boolean isConstant(IrSet s) {
076        IrDomain env = s.getEnv();
077        IrDomain ker = s.getKer();
078        assert IrUtil.isSubsetOf(ker, env);
079        return env.size() == ker.size();

103    public static boolean containsAll(int[] values, IrDomain domain) {
104        for (int value : values) {
105            if (!domain.contains(value)) {
106                return false;
107            }
108        }
109        return true;
110    }

112    public static boolean intersects(IrDomain d1, IrDomain d2) {
113        if (d1.isEmpty() || d2.isEmpty()) {
114            return false;
115        }
116        if (d1.getLowBound() > d2.getHighBound()) {
117            return false;
118        }
119        if (d1.getHighBound() < d2.getLowBound()) {
120            return false;
121        }
122        if (d1.isBounded() && d2.isBounded()) {
123            // Bounds are already checked.
124            return true;
125        }
126        if (d1.size() <= d2.size()) {
127            TIntIterator iter = d1.iterator();
128            while (iter.hasNext()) {
129                if (d2.contains(iter.next())) {
130                    return true;
131                }
132            }
133        } else {
134            TIntIterator iter = d2.iterator();
135            while (iter.hasNext()) {
136                if (d1.contains(iter.next())) {
137                    return true;
138                }
139            }
140        }
141        return false;
142    }

144    public static boolean isSubsetOf(IrDomain sub, IrDomain sup) {
145        if (sub.isEmpty()) {
146            return true;
147        }
148        if (sub.size() > sup.size()) {
149            return false;
150        }
151        if (sub.getLowBound() < sup.getLowBound()) {
152            return false;
153        }
154        if (sub.getHighBound() > sup.getHighBound()) {
155            return false;
156        }
157        if (sub.isBounded() && sup.isBounded()) {
158            // Bounds are already checked.
159            return true;
160        }
161        TIntIterator iter = sub.iterator();
162        while (iter.hasNext()) {
163            if (!sup.contains(iter.next())) {
164                return false;
165            }
166        }
167        return true;
168    }

170    public static IrDomain difference(IrDomain minuend, IrDomain subtrahend) {
171        if (!intersects(minuend, subtrahend)) {
172            return minuend;
173        }
174        if (minuend.isBounded() && subtrahend.isBounded()) {
175            if (minuend.getLowBound() < subtrahend.getLowBound()
176                    && minuend.getHighBound() <= subtrahend.getHighBound()) {
177                return Irs.boundDomain(minuend.getLowBound(), subtrahend.getLowBound() - 1);
178            }
179            if (minuend.getHighBound() > subtrahend.getHighBound()
180                    && minuend.getLowBound() >= subtrahend.getLowBound()) {
181                return Irs.boundDomain(subtrahend.getHighBound() + 1, minuend.getHighBound());
182            }
183        }
184        TIntHashSet values = new TIntHashSet();
185        values.addAll(minuend.getValues());
186        values.removeAll(subtrahend.getValues());
187        return Irs.enumDomain(values);
188    }

190    public static IrDomain intersection(IrDomain d1, IrDomain d2) {
191        if (isSubsetOf(d1, d2)) {
192            return d1;
193        }
194        if (isSubsetOf(d2, d1)) {
195            return d2;
196        }
197        if (d1.isBounded() && d2.isBounded()) {
198            if (d1.getLowBound() <= d2.getLowBound()
199                    && d1.getHighBound() >= d2.getLowBound()) {
200                return Irs.boundDomain(d2.getLowBound(), d1.getHighBound());
201            }
202            if (d2.getLowBound() <= d1.getLowBound()
203                    && d2.getHighBound() >= d1.getLowBound()) {
204                return Irs.boundDomain(d1.getLowBound(), d2.getHighBound());
205            }
206        }
207        TIntHashSet values = new TIntHashSet();
208        values.addAll(d1.getValues());
209        values.retainAll(d2.getValues());
210        return Irs.enumDomain(values);
211    }

213    public static IrDomain intersectionEnvs(IrSetExpr... sets) {
214        if (sets.length == 0) {
215            return Irs.EmptyDomain;
216        }
217        IrDomain domain = sets[0].getEnv();
218        for (int i = 1; i < sets.length; i++) {
219            domain = intersection(domain, sets[i].getEnv());
220        }
221        return domain;
222    }

224    public static IrDomain intersectionKers(IrSetExpr... sets) {
225        if (sets.length == 0) {
226            return Irs.EmptyDomain;
227        }
228        IrDomain domain = sets[0].getKer();
229        for (int i = 1; i < sets.length; i++) {
230            domain = intersection(domain, sets[i].getKer());
231        }
232        return domain;
233    }

235    public static IrDomain union(IrDomain d1, IrDomain d2) {
236        if (isSubsetOf(d1, d2)) {
237            return d2;
238        }
239        if (isSubsetOf(d2, d1)) {
240            return d1;
241        }
242        if (d1.isBounded() && d2.isBounded()) {
243            if (d1.getLowBound() <= d2.getLowBound()
244                    && d1.getHighBound() >= d2.getLowBound()) {
245                return Irs.boundDomain(d1.getLowBound(), d2.getHighBound());
246            }
247            if (d2.getLowBound() <= d1.getLowBound()
248                    && d2.getHighBound() >= d1.getLowBound()) {
249                return Irs.boundDomain(d2.getLowBound(), d1.getHighBound());
250            }
251        }
252        TIntHashSet values = new TIntHashSet();
253        values.addAll(d1.getValues());
254        values.addAll(d2.getValues());
255        return Irs.enumDomain(values);
256    }

263        for (int i = 1; i < sets.length; i++) {

274        for (int i = 1; i < sets.length; i++) {

280    public static IrDomain minus(IrDomain domain) {
281        if (domain.isEmpty()) {
282            return Irs.EmptyDomain;
283        }
284        if (domain.isBounded()) {
285            return Irs.boundDomain(-domain.getHighBound(), -domain.getLowBound());
286        }
287        int[] values = domain.getValues();
288        int[] minusValues = new int[values.length];
289        for (int i = 0; i < minusValues.length; i++) {
290            minusValues[i] = -values[i];
291        }
292        return Irs.enumDomain(minusValues);
293    }

295    public static IrDomain offset(IrDomain domain, int offset) {
296        if (domain.isEmpty()) {
297            return Irs.EmptyDomain;
298        }
299        if (domain.isBounded()) {
300            return Irs.boundDomain(domain.getLowBound() + offset, domain.getHighBound() + offset);
301        }
302        int[] values = domain.getValues();
303        int[] offsetValues = new int[values.length];
304        for (int i = 0; i < offsetValues.length; i++) {
305            offsetValues[i] = values[i] + offset;
306        }
307        return Irs.enumDomain(offsetValues);
308    }
