001package org.clafer.ir.analysis;
002
003import org.clafer.ir.IrRewriter;
004import java.util.ArrayList;
005import java.util.HashMap;
006import java.util.HashSet;
007import java.util.Iterator;
008import java.util.List;
009import java.util.Map;
010import java.util.Set;
011import org.clafer.ir.IrAnd;
012import org.clafer.ir.IrBoolExpr;
013import org.clafer.ir.IrCompare;
014import org.clafer.ir.IrDomain;
015import org.clafer.ir.IrImplies;
016import org.clafer.ir.IrIntExpr;
017import org.clafer.ir.IrModule;
018import org.clafer.ir.IrUtil;
019import static org.clafer.ir.Irs.*;
020
021/**
022 *
023 * @author jimmy
024 */
025public class Canonicalizer {
026
027    private Canonicalizer() {
028    }
029
030    /**
031     * Rewrites the module in a form that is easier to pattern match.
032     *
033     * @param module the module to canonicalize
034     * @return the canonicalized module
035     */
036    public static IrModule canonical(IrModule module) {
037        return module.withConstraints(canonicalizer.rewrite(and(module.getConstraints()), null));
038    }
039    private static final IrRewriter<Void> canonicalizer = new IrRewriter<Void>() {
040        @Override
041        public IrBoolExpr visit(IrAnd ir, Void a) {
042            Map<IrBoolExpr, Set<IrBoolExpr>> implications = new HashMap<IrBoolExpr, Set<IrBoolExpr>>();
043            List<IrBoolExpr> opts = new ArrayList<IrBoolExpr>(ir.getOperands().length);
044            for (IrBoolExpr operand : ir.getOperands()) {
045                IrBoolExpr opt = rewrite(operand, a);
046                if (opt instanceof IrImplies) {
047                    IrImplies implies = (IrImplies) opt;
048                    Set<IrBoolExpr> consequents = implications.get(implies.getAntecedent());
049                    if (consequents == null) {
050                        consequents = new HashSet<IrBoolExpr>();
051                        implications.put(implies.getAntecedent(), consequents);
052                    }
053                    consequents.add(implies.getConsequent());
054                } else {
055                    opts.add(opt);
056                }
057            }
058            for (Map.Entry<IrBoolExpr, Set<IrBoolExpr>> entry : implications.entrySet()) {
059                IrBoolExpr antecedent = entry.getKey();
060                Set<IrBoolExpr> consequents = entry.getValue();
061                if (!consequents.isEmpty()) {
062                    IrBoolExpr notAntecedent = not(antecedent);
063                    Set<IrBoolExpr> alternatives = implications.get(notAntecedent);
064                    if (alternatives != null && !alternatives.isEmpty()) {
065                        Iterator<IrBoolExpr> iter = consequents.iterator();
066                        while (iter.hasNext()) {
067                            IrBoolExpr consequent = iter.next();
068                            IrBoolExpr alternative = not(consequent);
069                            if (alternatives.remove(alternative)) {
070                                opts.add(
071                                        // Either choice is fine, but prefer to pick
072                                        // the antecedent that is not negative because
073                                        // it looks nicer and some cases easier to
074                                        // propagate.
075                                        antecedent.isNegative()
076                                        ? ifOnlyIf(notAntecedent, alternative)
077                                        : ifOnlyIf(antecedent, consequent));
078                                iter.remove();
079                            }
080                        }
081                    }
082                    for (IrBoolExpr consequent : consequents) {
083                        opts.add(implies(antecedent, consequent));
084                    }
085                    consequents.clear();
086                }
087            }
088            return and(opts);
089        }
090
091        @Override
092        public IrBoolExpr visit(IrCompare ir, Void a) {
093            IrIntExpr left = rewrite(ir.getLeft(), a);
094            IrIntExpr right = rewrite(ir.getRight(), a);
095            if (ir.getOp().isEquality()) {
096                IrDomain leftDomain = left.getDomain();
097                IrDomain rightDomain = right.getDomain();
098                // This canoicalizes the form somewhat so that is easier for the other
099                // analysis to find pattern matches.
100                if (leftDomain.size() == 2) {
101                    // Prefer the one that's closer to 0.
102                    int prefer = leftDomain.getLowBound();
103                    int unprefer = leftDomain.getHighBound();
104                    if (Math.abs(leftDomain.getLowBound()) > Math.abs(leftDomain.getHighBound())) {
105                        prefer = leftDomain.getHighBound();
106                        unprefer = leftDomain.getLowBound();
107                    }
108                    Integer rightConstant = IrUtil.getConstant(right);
109                    if (rightConstant != null && rightConstant.intValue() == unprefer) {
110                        return compare(left, ir.getOp().negate(), prefer);
111                    }
112                }
113                if (rightDomain.size() == 2) {
114                    // Prefer the one that's closer to 0.
115                    int prefer = rightDomain.getLowBound();
116                    int unprefer = rightDomain.getHighBound();
117                    if (Math.abs(rightDomain.getLowBound()) > Math.abs(rightDomain.getHighBound())) {
118                        prefer = rightDomain.getHighBound();
119                        unprefer = rightDomain.getLowBound();
120                    }
121                    Integer leftConstant = IrUtil.getConstant(left);
122                    if (leftConstant != null && leftConstant.intValue() == unprefer) {
123                        return compare(prefer, ir.getOp().negate(), right);
124                    }
125                }
126            }
127            return compare(left, ir.getOp(), right);
128        }
129    };
130}
