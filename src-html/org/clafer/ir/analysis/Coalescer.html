011import org.clafer.ir.IrBoolExpr;

023/**
024 * @author jimmy
025 */
026public class Coalescer {
027
028    private Coalescer() {
029    }
030
031    public static Pair<Map<IrIntVar, IrIntVar>, IrModule> coalesce(IrModule module) {
032        KeyGraph<IrIntVar> intGraph = new KeyGraph<IrIntVar>();
033        for (IrBoolExpr constraint : module.getConstraints()) {
034            if (constraint instanceof IrCompare) {
035                IrCompare compare = (IrCompare) constraint;
036                if (IrCompare.Op.Equal.equals(compare.getOp())
037                        && compare.getLeft() instanceof IrIntLiteral
038                        && compare.getRight() instanceof IrIntLiteral) {
039                    IrIntLiteral left = (IrIntLiteral) compare.getLeft();
040                    IrIntLiteral right = (IrIntLiteral) compare.getRight();
041                    intGraph.addEdge(left.getVar(), right.getVar());
042                    intGraph.addEdge(right.getVar(), left.getVar());
043                }
044            }
045            if (constraint instanceof IrIfOnlyIf) {
046                IrIfOnlyIf ifOnlyIf = (IrIfOnlyIf) constraint;
047                if (ifOnlyIf.getLeft() instanceof IrBoolCast
048                        && ifOnlyIf.getRight() instanceof IrBoolCast) {
049                    IrBoolCast left = (IrBoolCast) ifOnlyIf.getLeft();
050                    IrBoolCast right = (IrBoolCast) ifOnlyIf.getRight();
051                    if (left.getExpr() instanceof IrIntLiteral && right.getExpr() instanceof IrIntLiteral) {
052                        IrIntLiteral leftExpr = (IrIntLiteral) left.getExpr();
053                        IrIntLiteral rightExpr = (IrIntLiteral) right.getExpr();
054                        intGraph.addEdge(leftExpr.getVar(), rightExpr.getVar());
055                        intGraph.addEdge(rightExpr.getVar(), leftExpr.getVar());
056                    }
057                }
058            }
059        }
060        Map<IrIntVar, IrIntVar> coalescedInts = new HashMap<IrIntVar, IrIntVar>();
061        for (Set<IrIntVar> component : GraphUtil.computeStronglyConnectedComponents(intGraph)) {
062            if (component.size() > 1) {
063                Iterator<IrIntVar> iter = component.iterator();
064                IrIntVar var = iter.next();
065                StringBuilder name = new StringBuilder().append(var.getName());
066                IrDomain domain = var.getDomain();
067                while (iter.hasNext()) {
068                    var = iter.next();
069                    name.append(';').append(var.getName());
070                    domain = IrUtil.intersection(domain, var.getDomain());
071                }
072
073                if (domain.isEmpty()) {
074                    // Model is unsatisfiable. Compile anyways?
075                } else {
076                    IrIntVar coalesced = domainInt(name.toString(), domain);
077                    for (IrIntVar coalesce : component) {
078                        coalescedInts.put(coalesce, coalesced);
079                    }
080                }
081            }
082        }
083
084        IrIntVar[] coalescedIntVars = new IrIntVar[module.getIntVars().length];
085        for (int i = 0; i < coalescedIntVars.length; i++) {
086            IrIntVar intVar = module.getIntVars()[i];
087            IrIntVar coalescedIntVar = coalescedInts.get(intVar);
088            coalescedIntVars[i] = coalescedIntVar == null ? intVar : coalescedIntVar;
089        }
090        module.setIntVars(coalescedIntVars);
091        return new Pair<Map<IrIntVar, IrIntVar>, IrModule>(
092                coalescedInts,
093                new CoalesceRewriter(coalescedInts).rewrite(module, null));
094    }
095
096    private static class CoalesceRewriter extends IrRewriter<Void> {
097
098        private final Map<IrIntVar, IrIntVar> coalesced;
099
100        CoalesceRewriter(Map<IrIntVar, IrIntVar> coalesced) {
101            this.coalesced = coalesced;
102        }
103
104        @Override
105        public IrIntExpr visit(IrIntLiteral ir, Void a) {
106            IrIntVar var = coalesced.get(ir.getVar());
107            return var == null ? super.visit(ir, a) : $(var);
108        }
109    }
110}
