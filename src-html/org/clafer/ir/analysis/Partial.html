001package org.clafer.ir.analysis;
002
003import org.clafer.choco.constraint.OrConstraint;
004import solver.Solver;
005import solver.constraints.ICF;
006import solver.constraints.LCF;
007import solver.search.loop.monitors.IMonitorSolution;
008import solver.search.strategy.ISF;
009import solver.variables.IntVar;
010import solver.variables.VF;
011
012/**
013 *
014 * @author jimmy
015 */
016public class Partial {
017
018    public static void main(String[] args) {
019        Solver s = new Solver();
020        IntVar a = VF.enumerated("a", -2, 2, s);
021        IntVar b = VF.enumerated("b", -2, 2, s);
022        IntVar c = VF.enumerated("c", -2, 2, s);
023
024        s.getSearchLoop().plugSearchMonitor(new PP(s, a, b));
025        s.set(ISF.random(new IntVar[]{a, b}, System.nanoTime()));
026
027        if (s.findSolution()) {
028            do {
029                System.out.println(a.getValue() + " : " + b.getValue() + " : " + c.getValue());
030            } while (s.nextSolution());
031        }
032    }
033
034    private static final class PP implements IMonitorSolution {
035
036        private final Solver s;
037        private final IntVar a, b;
038
039        public PP(Solver s, IntVar a, IntVar b) {
040            this.s = s;
041            this.a = a;
042            this.b = b;
043        }
044
045        @Override
046        public void onSolution() {
047            s.postCut(
048                    new OrConstraint(
049                    ICF.arithm(a, "!=", a.getValue()), ICF.arithm(b, "!=", b.getValue())));
050        }
051    }
052}
