001package org.clafer.instance;
002
003import java.io.IOException;
004import org.clafer.common.Check;
005import org.clafer.ast.AstClafer;
006import org.clafer.ast.AstIntClafer;
007import org.clafer.ast.AstRef;
008import org.clafer.ast.AstUtil;
009
010/**
011 *
012 * @author jimmy
013 */
014public class InstanceClafer {
015
016    private final AstClafer type;
017    private final int id;
018    private final InstanceRef ref;
019    private final InstanceClafer[] children;
020
021    public InstanceClafer(AstClafer type, int id, InstanceRef ref, InstanceClafer... children) {
022        this.type = Check.notNull(type);
023        this.id = id;
024        this.ref = ref;
025        this.children = Check.noNulls(children);
026    }
027
028    public AstClafer getType() {
029        return type;
030    }
031
032    public int getId() {
033        return id;
034    }
035
036    public boolean hasRef() {
037        return ref != null;
038    }
039
040    public InstanceRef getRef() {
041        return ref;
042    }
043
044    public boolean hasChildren() {
045        return children.length != 0;
046    }
047
048    public InstanceClafer[] getChildren() {
049        return children;
050    }
051
052    /**
053     * Print solution to stdout.
054     *
055     * @throws IOException an IO error occurred
056     */
057    public void print() throws IOException {
058        print(System.out);
059    }
060
061    /**
062     * Print solution.
063     *
064     * @param out the stream to print to
065     * @throws IOException an IO error occurred
066     */
067    public void print(Appendable out) throws IOException {
068        print("", out);
069    }
070
071    private void print(String indent, Appendable out) throws IOException {
072        out.append(indent).append(type.getName()).append("#").append(Integer.toString(id));
073        if(hasRef()) {
074            out.append(" = ").append(ref.toString());
075        }
076//        AstRef typeRef = AstUtil.getInheritedRef(type);
077//        if (typeRef != null) {
078//            out.append(" = ");
079//            if (!(typeRef.getTargetType() instanceof AstIntClafer)) {
080//                out.append(typeRef.getTargetType().getName()).append("#");
081//            }
082//            out.append(Integer.toString(ref));
083//        }
084        out.append('\n');
085        for (InstanceClafer child : getChildren()) {
086            child.print(indent + "    ", out);
087        }
088    }
089
090    @Override
091    public String toString() {
092        StringBuilder result = new StringBuilder();
093        try {
094            print(result);
095        } catch (IOException e) {
096            // StringBuilder should not throw an IOException.
097            throw new Error(e);
098        }
099        return result.toString();
100    }
101}
